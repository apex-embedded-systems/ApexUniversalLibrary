573 static int64_t STX104_Fast_Time_Delta( struct timespec * ts_start, struct timespec * ts_end )
574 {
575  /* seconds, multiplied with 1 million */
576  int64_t usec_start = ts_start->tv_sec * 1000000;
577  int64_t usec_end = ts_end->tv_sec * 1000000;
578  /* Add full microseconds */
579  usec_start += ts_start->tv_nsec/1000;
580  usec_end += ts_end->tv_nsec/1000;
581  /* round up if necessary */
582  if (ts_start->tv_nsec % 1000 >= 500) ++usec_start;
583  if (ts_end->tv_nsec % 1000 >= 500) ++usec_end;
584  usec_end = usec_end - usec_start;
585  return usec_end;
