63 Initialization sequence:
64  1. turn off interrupts first, just in case they are active.
65  2. initialize triggering, make sure triggering has been stopped.
66  3. Initialize analog section
67  4. Initialize digital I/O section

70 the main points of initialization.

97  error_code = STX104_Trg_CFG_Set( board, &trg_cfg );
98  /* make sure trigger is now inactive/off */
99  error_code = STX104_Trg_Software_Stop( board );
100  /* analog inputs */
101  memset( &ai_cfg, 0, sizeof( stx104_ai_cfg_type ) ); /* clear configuration */
102  ai_cfg.gain = ACFG_GAIN_10V;
103  ai_cfg.burst_timer_ns = 5000; /* nS */
104  ai_cfg.frame_timer_ns = AI_FRAME_TIME_NS;

107  ai_cfg.last_channel = AI_LAST_CHANNEL;
108  ai_cfg.use_triggering_subsystem = true;
109  ai_cfg.high_speed_pre_queue_fifo = true;
110 #if defined( CONVERT_STRAIGHT_BINARY_TO_TWOS_COMPLEMENT )

468 int Demo_STX104_Main_Loop_Timing_Calculations( unsigned int time_msec, struct demo_stx104_data * ds )
469 {
470  double scratch;

472  ds->main_loop_time_sec = ( (double) time_msec ) * 1.0e-3; /* msec to sec */

474  time_sec = (long int) ds->main_loop_time_sec;
475  ds->main_loop_time_spec.tv_sec = time_sec;
476  scratch = ds->main_loop_time_sec - ( (double) time_sec );
477  scratch = scratch * 1.0e9; /* sec to nsec */
478  ds->main_loop_time_spec.tv_nsec = (long) scratch;
