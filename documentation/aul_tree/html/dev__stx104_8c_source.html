309 static int STX104_Reg_Read_Write( devmgr_device_type * device, io_xfr_type * xfr )
310 {
311  int error_code;
312  /* verify that command data width matches requested data width transfer */
313  if ( STX104_REG_GET_TYPE( xfr->reg_id ) != IO_XFR_GET_TYPE( xfr->command ) )
314  { error_code = EC_STX104_TRANSFER_DATA_TYPE_MISMATCH; goto STX104_Transfer_Error; }
315 
316  if ( STX104_REG_IS_INDEXED( xfr->reg_id ) )
317  { /* indexed register */
318  uint8_t index = (uint8_t)( STX104_REG_GET_OFFSET( xfr->reg_id ) );
319  error_code = SUCCESS;
320  switch( xfr->command )
321  {
322  case IO_XFR_WRITE_U8:
323  STX104_Index_Device_Write_U8( device, index, xfr->data.u8 );
324  break;
325  case IO_XFR_WRITE_U16:
326  STX104_Index_Device_Write_U16( device, index, xfr->data.u16 );
327  break;
328  case IO_XFR_WRITE_U32:
329  STX104_Index_Device_Write_U32( device, index, xfr->data.u32 );
330  break;
331  case IO_XFR_READ_U8:
332  STX104_Index_Device_Read_U8( device, index, &(xfr->data.u8) );
333  break;
334  case IO_XFR_READ_U16:
335  STX104_Index_Device_Read_U16( device, index, &(xfr->data.u16) );
336  break;
337  case IO_XFR_READ_U32:
338  STX104_Index_Device_Read_U32( device, index, &(xfr->data.u32) );
339  break;
340 /* TODO: move repeat functions to dev_stx104_index.c */
341  case IO_XFR_WRITE_U8_REPEAT:
342  {
343  size_t element;
344  size_t count_element_local = xfr->buf_len / sizeof( uint8_t );
345  for ( element = 0; element < count_element_local; element++ )
346  {
347  STX104_Index_Device_Write_U8( device, index, ((uint8_t *) xfr->data.buf)[element] );
348  if ( error_code ) goto STX104_Transfer_Error;
349  }
350  }
351  break;
352  case IO_XFR_WRITE_U16_REPEAT:
353  {
354  size_t element;
355  size_t count_element_local = xfr->buf_len / sizeof( uint16_t );
356  for ( element = 0; element < count_element_local; element++ )
357  {
358  STX104_Index_Device_Write_U16( device, index, ((uint16_t *) xfr->data.buf)[element] );
359  if ( error_code ) goto STX104_Transfer_Error;
360  }
361  }
362  break;
363  case IO_XFR_WRITE_U32_REPEAT:
364  {
365  size_t element;
366  size_t count_element_local = xfr->buf_len / sizeof( uint32_t );
367  for ( element = 0; element < count_element_local; element++ )
368  {
369  STX104_Index_Device_Write_U32( device, index, ((uint32_t *) xfr->data.buf)[element] );
370  if ( error_code ) goto STX104_Transfer_Error;
371  }
372  }
373  break;
374  case IO_XFR_READ_U8_REPEAT:
375  {
376  size_t element;
377  size_t count_element_local = xfr->buf_len / sizeof( uint8_t );
378  for ( element = 0; element < count_element_local; element++ )
379  {
380  STX104_Index_Device_Read_U8( device, index, &( ((uint8_t *) xfr->data.buf)[element] ) );
381  if ( error_code ) goto STX104_Transfer_Error;
382  }
383  }
384  break;
385  case IO_XFR_READ_U16_REPEAT:
386  {
387  size_t element;
388  size_t count_element_local = xfr->buf_len / sizeof( uint16_t );
389  for ( element = 0; element < count_element_local; element++ )
390  {
391  STX104_Index_Device_Read_U16( device, index, &( ((uint16_t *) xfr->data.buf)[element] ) );
392  if ( error_code ) goto STX104_Transfer_Error;
393  }
394  }
395  break;
396  case IO_XFR_READ_U32_REPEAT:
397  {
398  size_t element;
399  size_t count_element_local = xfr->buf_len / sizeof( uint32_t );
400  for ( element = 0; element < count_element_local; element++ )
401  {
402  STX104_Index_Device_Read_U32( device, index, &( ((uint32_t *) xfr->data.buf)[element] ) );
403  if ( error_code ) goto STX104_Transfer_Error;
404  }
405  }
406  break;
407  default:
408  break;
409  }
410  goto STX104_Transfer_Continue;
411  }
412  else
413  { /* direct address */
414  xfr->offset = STX104_REG_GET_OFFSET( xfr->reg_id ) + device->io_region_offset;
415  }
416  error_code = IO_Ioctl( device->io_device, CMD_IO_XFR, (unsigned long)((void *) xfr) );
417 STX104_Transfer_Continue:
418  if ( error_code ) goto STX104_Transfer_Error;
419  return SUCCESS;
420 STX104_Transfer_Error:
421  if ( error_code < 0 ) error_code = -error_code;
422  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
423  return -error_code;
424 }

580 static void STX104_SVN_Time_Stamp_Merge( void )
581 {
582  uint32_t version;
583  const char * version_str;
584  uint32_t max_version;
585  size_t index;
586  const char * max_str;
587  /* determine which time stamp is the latest */
588  index = 0;
589  max_version = 0;
590  max_str = NULL;
591  do
592  {
593  version_str = ( * stx104_ts_fnc_list[index] )();
594  version = TSSVN_YYMMDDHH_DEC( version_str );
595  if ( version > max_version )
596  {
597  max_version = version;
598  max_str = ( * stx104_ts_fnc_list[index] )();
599  }
