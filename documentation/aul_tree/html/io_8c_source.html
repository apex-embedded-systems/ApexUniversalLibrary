353 static int IO_Xfr_Command( int iod, unsigned long argument )
354 {
355  int error_code;
356  size_t address_limit;
357  io_xfr_type xfr;
358 #if defined( IO_DEBUG_LOG )
359  Debug_Log( IO_DEBUG_LOG "%s: entry\n", __func__ );
360 #endif
361  /* get transfer command from user space */
362  if ( IO_COPY_FROM_USER( &xfr, argument, sizeof( xfr ) ) )
363  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
364  /* Test the range of the virtual_address to make sure it is within
365  the devices range.
366  */
367  address_limit = io_ds[iod].io_definition.region_offset;
368  if ( xfr.offset < address_limit /* lower end */ )
369  { error_code = EC_IO_ADDRESS_RANGE; goto IO_Xfr_Command_Error_Return_To_User; }
370  address_limit += io_ds[iod].io_definition.region_bytes;
371  if ( xfr.offset >= address_limit /* upper end */ )
372  { error_code = EC_IO_ADDRESS_RANGE; goto IO_Xfr_Command_Error_Return_To_User; }
373 
374  error_code = SUCCESS;
375  switch( xfr.command )
376  {
377  case IO_XFR_WRITE_U8:
378  IO_Write_U8( iod, xfr.offset, xfr.data.u8 );
379  break;
380  case IO_XFR_WRITE_U16:
381  IO_Write_U16( iod, xfr.offset, xfr.data.u16 );
382  break;
383  case IO_XFR_READ_U8:
384  IO_Read_U8( iod, xfr.offset, &(xfr.data.u8) );
385  break;
386  case IO_XFR_READ_U16:
387  IO_Read_U16( iod, xfr.offset, &(xfr.data.u16) );
388  break;
389 
390 /* TODO: all repeat functions to be merged to reduce code duplication. */
391 
392  case IO_XFR_WRITE_U8_REPEAT:
393  {
394  size_t index = 0;
395  size_t count_element = xfr.buf_len; /* count in elements */
396  size_t count_u8 = IO_BUFFER_LOCAL_SIZE; /* buffer size in bytes */
397  size_t count_element_local = IO_BUFFER_LOCAL_SIZE / sizeof( uint8_t );
398  while ( count_element > count_element_local )
399  {
400  if ( IO_COPY_FROM_USER( io_buffer_local, &(xfr.data.buf[index * count_u8]), count_u8 ) )
401  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
402  IO_Write_Repeat_U8( iod, xfr.offset, (uint8_t *) io_buffer_local, count_element_local );
403  count_element = count_element - count_element_local;
404  index++;
405  }
406  count_u8 = count_element * sizeof( uint8_t );
407  if ( IO_COPY_FROM_USER( io_buffer_local, &(xfr.data.buf[index * count_u8]), count_u8 ) )
408  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
409  IO_Write_Repeat_U8( iod, xfr.offset, (uint8_t *) io_buffer_local, count_element );
410  }
411  break;
412  case IO_XFR_WRITE_U16_REPEAT:
413  {
414  size_t index = 0;
415  size_t count_element = xfr.buf_len; /* count in elements */
416  size_t count_u8 = IO_BUFFER_LOCAL_SIZE; /* buffer size in bytes */
417  size_t count_element_local = IO_BUFFER_LOCAL_SIZE / sizeof( uint16_t );
418  while ( count_element > count_element_local )
419  {
420  if ( IO_COPY_FROM_USER( io_buffer_local, &(xfr.data.buf[index * count_u8]), count_u8 ) )
421  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
422  IO_Write_Repeat_U16( iod, xfr.offset, (uint16_t *) io_buffer_local, count_element_local );
423  count_element = count_element - count_element_local;
424  index++;
425  }
426  count_u8 = count_element * sizeof( uint16_t );
427  if ( IO_COPY_FROM_USER( io_buffer_local, &(xfr.data.buf[index * count_u8]), count_u8 ) )
428  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
429  IO_Write_Repeat_U16( iod, xfr.offset, (uint16_t *) io_buffer_local, count_element );
430  }
431  break;
432  case IO_XFR_READ_U8_REPEAT:
433  {
434  size_t index = 0;
435  size_t count_element = xfr.buf_len; /* count in elements */
436  size_t count_u8 = IO_BUFFER_LOCAL_SIZE; /* buffer size in bytes */
437  size_t count_element_local = IO_BUFFER_LOCAL_SIZE / sizeof( uint8_t );
438  while ( count_element > count_element_local )
439  {
440  IO_Read_Repeat_U8( iod, xfr.offset, (uint8_t *) io_buffer_local, count_element_local );
441  if ( IO_COPY_TO_USER( &(xfr.data.buf[index * count_u8]), io_buffer_local, count_u8 ) )
442  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
443  count_element = count_element - count_element_local;
444  index++;
445  }
446  IO_Read_Repeat_U8( iod, xfr.offset, (uint8_t *) io_buffer_local, count_element );
447  count_u8 = count_element * sizeof( uint8_t );
448  if ( IO_COPY_TO_USER( &(xfr.data.buf[index * count_u8]), io_buffer_local, count_u8 ) )
449  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
450  }
451  break;
452  case IO_XFR_READ_U16_REPEAT:
453  {
454  size_t index = 0;
455  size_t count_element = xfr.buf_len; /* count in elements */
456  size_t count_u8 = IO_BUFFER_LOCAL_SIZE; /* buffer size in bytes */
457  size_t count_element_local = IO_BUFFER_LOCAL_SIZE / sizeof( uint16_t );
458  while ( count_element > count_element_local )
459  {
460  IO_Read_Repeat_U16( iod, xfr.offset, (uint16_t *) io_buffer_local, count_element_local );
461  if ( IO_COPY_TO_USER( &(xfr.data.buf[index * count_u8]), io_buffer_local, count_u8 ) )
462  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
463  count_element = count_element - count_element_local;
464  index++;
465  }
466  IO_Read_Repeat_U16( iod, xfr.offset, (uint16_t *) io_buffer_local, count_element );
467  count_u8 = count_element * sizeof( uint16_t );
468  if ( IO_COPY_TO_USER( &(xfr.data.buf[index * count_u8]), io_buffer_local, count_u8 ) )
469  { error_code = EC_IO_COPY_FROM_USER; goto IO_Xfr_Command_Error; }
470  }
471  break;
472  default:
473  break;
474  }
475  /* return results to user space */
476  xfr.error_code = error_code;
477  if ( IO_COPY_TO_USER( argument, &xfr, sizeof( xfr ) ) )
478  { error_code = EC_IO_COPY_TO_USER; goto IO_Xfr_Command_Error; }
479 
480 #if defined( IO_DEBUG_LOG )
481  Debug_Log( IO_DEBUG_LOG "%s: exit success\n", __func__ );
482 #endif
483  return SUCCESS;
484 IO_Xfr_Command_Error_Return_To_User:
485  /* return results to user space */
486  xfr.error_code = error_code;
487  if ( IO_COPY_TO_USER( argument, &xfr, sizeof( xfr ) ) )
488  { error_code = EC_IO_COPY_TO_USER; }
489 IO_Xfr_Command_Error:
490  if ( error_code < 0 ) error_code = -error_code;
491  Error_Log( module_io_definition.name, __func__, __LINE__, error_code );
492  io.last_error_code = error_code;
493 #if defined( IO_DEBUG_LOG )
494  Debug_Log( IO_DEBUG_LOG "%s: exit ec=%s\n", __func__, Status_EC_String( error_code ) );
495 #endif
496  return -error_code;
497 }

633 int IO_Ioctl( int iod, unsigned int command, unsigned long argument )

643  /* Is the command intended for a module? */
644 /* if ( IOC_INVALID_LOGICAL_ID_MODULE( module_logical_id ) ) */
