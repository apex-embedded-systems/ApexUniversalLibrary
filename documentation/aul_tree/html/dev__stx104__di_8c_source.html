144 AUL_DLL_API int AULCALLCONV STX104_DI_Get( int board, STX104_DI_ENUM channel, BOOL * value )
145 {
146  int error_code;
147  devmgr_device_type * device;
148  size_t address;
149  uint8_t data_di;
150  uint8_t mask;
151  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_DI_Get_Error;
152  if ( (channel >= STX104_DI0) && (channel <= STX104_DI3) )
153  {
154  mask = 0x01 << channel;
155  address = device->io_region_offset + STX104_DIGITAL_INPUTS;
156  IO_Read_U8( device->io_device, address, &data_di );
157  if ( data_di & mask ) *value = true;
158  else *value = false;
159  }
160  else
161  {
162  error_code = EC_STX104_CHANNEL_OUT_OF_RANGE; goto STX104_DI_Get_Error;
163  }
164  return error_code;
165 STX104_DI_Get_Error:
166  if ( error_code < 0 ) error_code = -error_code;
