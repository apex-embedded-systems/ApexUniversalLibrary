94 
95  /* calculate the indexed address location */
96  uc_reg_offset = uc_reg->virtual_address + upper_component->offset;
97 
98  if ( REG_TEST_AUTO_INCREMENT( uc_reg->info ) )
99  { /* write to sequential locations */
100  if ( REG_BYTES_LESS_ONE(1) == REG_GET_REGISTER_APERTURE( uc_reg->info ) )

109  ar_size
110  );
111

113  CmpMgr_Reg_Write( iac,
114  INDIRECT_DATA_LSB, /* the data register */
115  &(buf[i]), /* the data to be written to the register */
116  sizeof(uint8_t)
117  );
118  }
119  }
120  else
121  { /* write out 16-bit word at a time, address is byte resolution, size does not matter */
122  buf_u16 = (uint16_t *) buf; /* char to UINT16 */
123  u16_total = buf_len / sizeof( uint16_t ); /* total UINT16 count */

131  ar_size

137  (char *)(&(buf_u16[i])),/* the data to be written to the register */
138  sizeof(uint16_t)
139  );
140  }
141  }
142  }
143  else
144  { /* write to same location 'element_count' times */
145  if ( REG_BYTES_LESS_ONE(1) == REG_GET_REGISTER_APERTURE( uc_reg->info ) )
146  { /* write out single byte at a time */
147  if ( REG_SIZE_UINT8 == uc_reg_size )
148  { /* write out 8-bit byte string */
149  for ( i = 0; i < buf_len; i++ )
150  {
151  address = uc_reg_offset;
152  /* write the address out first */
153  CmpMgr_Reg_Write( iac,
154  INDIRECT_INDEX, /* the address register */

156  ar_size
157  );
158

162  &(buf[i]), /* the data to be written to the register */
163  sizeof(uint8_t)
164  );
165  }
166  }
167  else
168  { /* write words a byte at a time */
169  element_total = buf_len / uc_reg_size;
170  for ( i = 0; i < element_total; i = i + uc_reg_size )
171  {
172  for( word = 0; word < uc_reg_size; word++ )
173  {
174  address = uc_reg_offset + word;
175  /* write the address out first */
176  CmpMgr_Reg_Write( iac,
177  INDIRECT_INDEX, /* the address register */
178  (char *)(&address), /* the address of the register to be written */
179  ar_size
180  );
181

183  CmpMgr_Reg_Write( iac,
184  INDIRECT_DATA, /* the data register */
185  &(buf[i + word]), /* the data to be written to the register */
186  sizeof(uint8_t)
187  );
188  }
189  }
190  }
191  }
192  else
193  { /* write out 16-bit word at a time */
194  /* write 16-bit word string, address is byte resolution, size matters! */
195  buf_u16 = (uint16_t *) buf;
196  u16_total = buf_len / sizeof( uint16_t );
197  if ( (buf_len & 0x0001) == buf_len )
198  { /* odd number of bytes */
199  u16_total++;
200  }

202  { /* size is UINT8 or UINT16. */
203  for ( i = 0; i < u16_total; i++ )
204  {

206  /* write the address out first */
207  CmpMgr_Reg_Write( iac,
208  INDIRECT_INDEX, /* the address register */
209  (char *)(&address), /* the address of the register to be written */
210  ar_size
211  );
212

217  sizeof(uint16_t)
218  );
219  }
220  }
