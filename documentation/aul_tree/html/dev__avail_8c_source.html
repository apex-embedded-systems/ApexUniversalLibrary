107 int Device_Available_Find( IOC_T logical_id, device_definition_type ** definition_ptr_out )
108 {
109  size_t index;
110  IOC_T logical_id_from_list;
111  index = 0;
112  /* make sure that we just have the logical_id, not a physical_id */
113  logical_id = IOC_GET_LOGICAL_ID_DEVICE( logical_id );
114  if ( IOC_INVALID_LOGICAL_ID_DEVICE( logical_id ) ) return -1;
115  while ( NULL != device_available_list[index].definition )
116  {
117  logical_id_from_list = IOC_GET_LOGICAL_ID_DEVICE( device_available_list[index].definition->io_definition.physical_id );
118  if ( logical_id == logical_id_from_list )
119  {
120  *definition_ptr_out = device_available_list[index].definition;
121  return SUCCESS;
122  }
123  index++;
124  }
125  *definition_ptr_out = NULL;
126  return -1; /*TODO: ??*/
127 }
