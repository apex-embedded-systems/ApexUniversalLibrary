50 Description
51 Reads data from hardware I/O space given register parameters.
52 It takes into account the targets actual bus width as specified
53 in the target.h file. The register aperture maybe different than
54 the bus aperture, so it is the responsibility of this function to
55 determine the actual aperture; whichever is narrower takes precedence.
56 
57 This function is capable of reading one or more elements to a given
58 register.
59 
60 Assumptions
61 1. Register actually has read access.
62 2. Address is byte resolution
63 3. Actual address location that is calculated is a valid readable location.
64 */
65 static void Reg_IO_Direct_Read( REGISTER_TYPE * reg,
66  size_t offset,
67  int iod,
68  char * buf,
69  size_t buf_len
70  )
71 {
72  size_t address;
73  size_t element_total;
74  size_t u16_total;
75  size_t index;
76  size_t element_index;
77  size_t size_reg;
78  uint16_t * buf_u16;
79  size_t bus_aperture;
80  size_t reg_aperture;
81  const size_t u16_size = sizeof( uint16_t );
82 
83  /* offset is a base address adjustment, i.e. multiple UARTs with same register_tables */
84  address = reg->virtual_address + offset;
85 
86 #if defined( REG_DEBUG )
87  if ( REG_DIR_NONE == ( REG_GET_DIRECTION( reg->info ) & REG_DIR_READ ) )
88  {
89  Debug_Log( REG_DEBUG_LOG "%s: cannot read \"%s\" at 0x%04X\n", __func__, reg->name, address );
90  return;
91  }
92 #endif
93 
94 /* TODO: this is not really where I want this, but for now it is OK. */
95  bus_aperture = REG_BYTES_LESS_ONE( TARGET_BUS_APERTURE_WIDTH_BYTES );
96  reg_aperture = REG_GET_REGISTER_APERTURE( reg->info );
97  if ( reg_aperture > bus_aperture ) reg_aperture = bus_aperture;
98 
99  if ( REG_TEST_AUTO_INCREMENT( reg->info ) )
100  { /***
101  Auto-Increment Read Operations (AIRO)
102  Read sequential byte or word locations, incrementing the address as we go.
103  Size of the actual elements in the buffer do not matter. In this case the
104  register or bus aperture are the primary drivers in how the data is written.
105  NOTE: this portion has been used rarely.
106  */
107  if ( REG_BYTES_LESS_ONE(1) == reg_aperture )
108  { /* 8-BIT AIRO -- begin */
109  for ( index = 0; index < buf_len; index++ )
110  {
111  IO_Read_U8( iod, address + index, (uint8_t *) &(buf[index]) );
112  }
113  } /* 8-BIT AIRO -- end */
114  else
115  { /* 16-BIT AIRO -- begin */
116  buf_u16 = (uint16_t *) buf; /* char to uint16_t */
117  u16_total = buf_len / u16_size; /* total uint16_t count */
118  for ( index = 0; index < u16_total; index++ )
119  {
120  IO_Read_U16( iod, address + index * u16_size, &(buf_u16[index]) );
121  }
122  } /* 16-BIT AIRO -- end */
123  }
124  else
125  { /***
126  Register Read Operations (RRO)
127  Read from the same register location. The address is only incremented
128  to support reads from specific register. This supports multiple
129  element read operations (i.e. string reads).
130  */
131  size_reg = REG_GET_ELEMENT_SIZE( reg->info );
132  if ( REG_BYTES_LESS_ONE(1) == reg_aperture )
133  { /* Register aperture is restricted to 8-bits */
134  if ( REG_SIZE_UINT8 == size_reg )
135  { /* 8-BIT REGISTER RRO -- begin */
136  IO_Read_Repeat_U8( iod, address, (uint8_t *)buf, buf_len );
137  } /* 8-BIT REGISTER RRO -- end */
138  else
139  { /* 16-BIT REGISTER RRO w/8-bit restrict -- begin */
140  element_total = buf_len / size_reg;
141  for ( element_index = 0; element_index < element_total; element_index++ )
142  {
143  for( index = 0; index < u16_size; index++ )
144  {
145  /*buf[index + word] = IOKERN_READ_uint8_t( address + word );*/
146  IO_Read_U8( iod, address + index, (uint8_t *) &(buf[element_index*u16_size + index]) );
147  }
148  }
149  } /* 16-BIT REGISTER RRO w/8-bit restrict -- end */
150  }
151  else
152  { /* 16-BIT RRO - one or more elements to register location -- begin */
153  buf_u16 = (uint16_t *) buf;
154  element_total = buf_len / u16_size; /* fixed to 16-bits */
155  if ( REG_SIZE_UINT8 == size_reg )
156  { /* 8-bit data to 16-bit width path. Register is uint8_t and will read LSB only */
157  uint16_t value_u16;
158  for ( element_index = 0; element_index < element_total; element_index++ )
159  {
160  IO_Read_U16( iod, address, &value_u16 );
161  buf[element_index] = (uint8_t)( value_u16 & UINT8_MAX );
162  }
163  }
164  else if ( REG_SIZE_UINT16 == size_reg )
165  { /* Register is uint16_t in size -- read element(s) directly */
166  IO_Read_Repeat_U16( iod, address, buf_u16, element_total );
167  }
168  else
169  { /* Register is uint32_t or larger -- write 16-bits at a time */
170  element_total = buf_len / size_reg;
171  /* number of 16-bit writes for given register */
172  u16_total = size_reg / u16_size;
173  for ( element_index = 0; element_index < element_total; element_index++ )
174  {
175  for( index = 0; index < u16_total; index++ )
176  {
177  IO_Read_U16( iod,
178  address + index * u16_size,
179  &(buf_u16[element_index * u16_size + index] )
180  );
181  }
182  }
183  }
184  } /* 16-BIT RRO - one or more elements to register location -- end */
185  }
