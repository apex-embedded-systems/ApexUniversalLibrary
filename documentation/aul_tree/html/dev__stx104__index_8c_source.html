49 void STX104_Index_Bank_Set( devmgr_device_type * device, uint8_t bank /* 0 or 1 */ )
50 {
51  size_t address;
52  uint8_t scratch;
53  uint8_t value;
54 
55  address = device->io_region_offset + STX104_ADC_CONFIGURATION;
56  IO_Read_U8( device->io_device, address, &scratch );
57  if ( bank == 0 )
58  {
59  if ( (scratch & 0x80) == 0x80 )
60  {
61  scratch = scratch & 0x0F;
62  value = scratch | 0x30;
63  IO_Write_U8( device->io_device, address, value );
64  value = scratch | 0x30;
65  IO_Write_U8( device->io_device, address, value );
66  value = scratch | 0xB0;
67  IO_Write_U8( device->io_device, address, value );
68  value = scratch | 0xA0;
69  IO_Write_U8( device->io_device, address, value );
70  }
71  }
72  else
73  {
74  if ( (scratch & 0x80) == 0 )
75  {
76  scratch = scratch & 0x0F;
77  value = scratch | 0xF0;
78  IO_Write_U8( device->io_device, address, value );
79  value = scratch | 0xF0;
80  IO_Write_U8( device->io_device, address, value );
81  value = scratch | 0x50;
82  IO_Write_U8( device->io_device, address, value );
83  value = scratch | 0xD0;
84  IO_Write_U8( device->io_device, address, value );
85  }
86  }
87 }
