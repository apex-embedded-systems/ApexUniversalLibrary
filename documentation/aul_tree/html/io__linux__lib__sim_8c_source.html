286 static int IO_LLS_Ioctl( struct io_device_dataset * ds, unsigned int command, unsigned long argument )
287 {
288  int error_code;
289  (void) ds;
290  (void) command;
291  (void) argument;
292 #if defined( IO_DEBUG_LOG )
293  Debug_Log( IO_DEBUG_LOG "%s: entry\n", __func__ );
294 #endif
295  error_code = EC_IO_IOCTL_COMMAND_UNKNOWN; /* command invalid */
296  switch( command )
297  { /* emulate the driver */
298  case CMD_AULDRV_NAME:
299  {
300  if ( IO_COPY_TO_USER( argument, io_target_definition.name, String_Length( io_target_definition.name ) + 1 ) )
301  { error_code = EC_IO_COPY_TO_USER; goto IO_LLS_Ioctl_Error; }
302  else
303  { error_code = SUCCESS; }
304  }
305  break;
306  case CMD_AULDRV_DEVICE_LIST_GET:
307  { /* copy the io_definitions into place */
308  device_list_type device_list;
309  size_t count;
310  Mem_Set( &device_list, 0, sizeof( device_list ) );
311  count = 0;
312  while( IOC_PHYSICAL_ID_NULL != io_lls_definition[count].physical_id )
313  {
314  Mem_Copy( &(device_list.region[count]), &(io_lls_definition[count]), sizeof( struct io_definition ) );
315  count++;
316  }
317  device_list.region_count = count;
318  device_list.type = SYS_TYPE_MAKE( device_list_type );
319  if ( IO_COPY_TO_USER( argument, &device_list, sizeof(device_list) ) )
320  { error_code = EC_IO_COPY_TO_USER; goto IO_LLS_Ioctl_Error; }
321  }
322  return -EC_IO_WARN_DEVICE_LIST_PARTIAL;
323  case CMD_AULDRV_DEBUG_ACTIVE_GET:
324  { /* return zero if IO_DEBUG is not defined, otherwise non-zero */
325  int debug_active;
326 #ifdef IO_DEBUG
327  debug_active = 1;
328 #else
329  debug_active = 0;
330 #endif
331  if ( IO_COPY_TO_USER( argument, &debug_active, sizeof( debug_active ) ) )
332  { error_code = EC_AULDRV_COPY_TO_USER; goto IO_LLS_Ioctl_Error; }
333  }
334  break;
335  }
336  if ( error_code ) goto IO_LLS_Ioctl_Error;
337 #if defined( IO_DEBUG_LOG )
338  Debug_Log( IO_DEBUG_LOG "%s: exit success\n", __func__ );
339 #endif
340  return SUCCESS;
341 IO_LLS_Ioctl_Error:
342  Error_Log( MODULE_IO_NAME, __func__, __LINE__, error_code );
343 #if defined( IO_DEBUG_LOG )
344  Debug_Log( IO_DEBUG_LOG "%s: exit ec=%s\n", __func__, Status_EC_String( error_code ) );
345 #endif
346  return error_code;
347 }
