125  /* defaults */

131  #endif
132  fmt = format;

134  if ( index >= 3 )
135  {
136  if ( (format[0] == '<') && (format[2] == '>') )
137  {
138  switch( format[1] )

141  #if !defined( DEBUG_KERNEL )

143  #endif

146  case 'f': /* append to file */
147  #if !defined( DEBUG_KERNEL )

152  }

157  #endif

159  break;
160  }
161  }
162  }

164  #ifdef DEBUG_PREPEND_DEBUG
