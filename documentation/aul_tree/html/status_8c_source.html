181 const char * Status_EC_String( int error_code )
182 {
183  size_t index;
184  if ( SUCCESS == error_code ) return status_success_name;
185  if ( error_code < 0 ) error_code = -error_code;
186 
187  if ( EC_ID_IS_OS( error_code ) )
188  {
189 #if defined( _WIN32 )
190 /* http://stackoverflow.com/questions/1387064/how-to-get-the-error-message-from-the-error-code-returned-by-getlasterror */
191  Status_String_Win32_Get( error_code );
192  return (const char *) status_string_buf;
193 #elif defined( __linux__ )
194  #ifdef __KERNEL__
195  return status_zero_length_string; /* no strerror() within kernel */
196  #else
197  return strerror( error_code );
198  #endif
199 #else
200  return NULL;
201 #endif
202  }
203  else
204  {
205  index = 0;
206  while( 0 != status_ds_list[index].ec )
207  {
208  if ( ((size_t) error_code) == status_ds_list[index].ec )
209  {
210  return status_ds_list[index].name;
211  }
212  index++;
213  }
214  }
215  return NULL;
216 }

362 int Status_CSV_Column_Width( STATUS_CSV_COLUMN_ENUM column )
363 {
364  const char * string_pointer;
365  size_t width, width_max;
366  size_t row_count;
367  size_t row;
368 #define STATUS_CW_LEN 32
369  char buf[STATUS_CW_LEN];
370 
371  if ( ( column >= STATUS_CSV_COLUMN_UNKNOWN ) ||
372  ( column < STATUS_CSV_COLUMN_ENUM_VALUE ) ) goto Status_CSV_Column_Width_Error;
373 
374  row_count = Status_All_Count();
375  width_max = 0;
376  for ( row = 0; row < row_count; row++ )
377  {
378  switch( column )
379  {
380  case STATUS_CSV_COLUMN_ENUM_VALUE:
381  Mem_Set( buf, 0, STATUS_CW_LEN );
382  STATUS_SNPRINTF( buf, STATUS_CW_LEN, "%d", status_ds_list[row].ec );
383  if ( '\0' != buf[STATUS_CW_LEN - 1] )
384  { buf[STATUS_CW_LEN - 1] = '\0'; goto Status_CSV_Column_Width_Error; }
385  string_pointer = buf;
386  break;
387  case STATUS_CSV_COLUMN_ENUM_SYMBOL:
388  string_pointer = status_ds_list[row].ec_name;
389  break;
390  case STATUS_CSV_COLUMN_MESSAGE:
391  string_pointer = status_ds_list[row].name;
392  break;
393  case STATUS_CSV_COLUMN_MODULE_NAME:
394  {
395  size_t id = EC_GET_ID( status_ds_list[row].ec );
396  string_pointer = Module_Name( id );
397  }
398  break;
399  default:
400  string_pointer = NULL;
401  break;
402  }
403  if ( NULL != string_pointer )
404  {
405  width = String_Length( string_pointer );
406  if ( width > width_max ) width_max = width;

410 Status_CSV_Column_Width_Error:
