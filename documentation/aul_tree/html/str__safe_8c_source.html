75 Description
76 Non-case sensitive string compare. Requires valid string pointers.
77 the str_a_size is the maximum storage at str_a including null character.
78 This compare is case independent.
79 */
80 int String_Compare( const char * str_a, const char * str_b, int str_a_size )
81 {
82  int index;
83  char char_a, char_b;
84  int difference;
85  int term_a; /* terminate (worse case) on str_a null */
86 
87  if ( 0 == str_a_size ) return 0;
88  if ( ( NULL == str_a ) && ( NULL == str_b ) ) return 0;
89  if ( NULL == str_a ) return -str_a_size; /* not sure best number */
90  if ( NULL == str_b ) return str_a_size; /* not sure best number */
91 
92  if ( STRING_STR_A_NULL_TERMINATED == str_a_size ) term_a = 1;
93  else term_a = 0;
94 
95  index = 0;
96  do
97  {
98  char_a = str_a[index];
99  char_b = str_b[index];
100 
101  if ( char_a ) char_a = char_a | ASCII_LOWER_CASE_MASK;
102  if ( char_b ) char_b = char_b | ASCII_LOWER_CASE_MASK;
103 
104  difference = ((unsigned char) char_a) - ((unsigned char) char_b);
105  index++;
106  } while ( ( 0 == difference ) &&
107  ( ASCII_NULL != char_a ) &&
108  ( ASCII_NULL != char_b ) &&
109  (( index < str_a_size ) || term_a) ); /* was index <= str_a_size */
110 
111  return difference;
112 }
