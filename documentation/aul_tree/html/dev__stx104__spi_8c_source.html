48 int STX104_SPI_DI_Config_Request_Check( int board, STX104_DI_CFG_TYPE * di_cfg_request, const char * prefix )
49 {
50  int error_code;
51  devmgr_device_type * device;
52  STX104_DI_CFG_TYPE di_cfg; /* current configuration */
53  STX104_SPI_CFG_TYPE spi_cfg;
54  stx104_dataset_type * ds;
55  (void) prefix;
56  error_code = SUCCESS;
57  device = AUL_Device_Get( board );
58  ds = (stx104_dataset_type *) device->private_data;
59  if ( !(ds->spi_interface_active) ) return SUCCESS;
60  /* So long as the new request does not affect the digital
61  input mapped by SDI_MAP we are OK.
62  */
63  STX104_DI_CFG_Get( board, &di_cfg );
64  if ( di_cfg_request->deglitch_200ns != di_cfg.deglitch_200ns )
65  { error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR; goto STX104_SPI_DI_Config_Request_Check_Error; }
66 
67  STX104_SPI_CFG_Get( board, &spi_cfg );
68  switch( spi_cfg.sdi_map )
69  {
70  case SPI_SDI_DIN0:
71  if ( di_cfg_request->polarity.din0 != di_cfg.polarity.din0 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
72  break;
73  case SPI_SDI_DIN1:
74  if ( di_cfg_request->polarity.din1 != di_cfg.polarity.din1 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
75  if ( di_cfg_request->swap_din1_and_din3 != di_cfg.swap_din1_and_din3 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
76  break;
77  case SPI_SDI_DIN2:
78  if ( di_cfg_request->polarity.din2 != di_cfg.polarity.din2 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
79  break;
80  case SPI_SDI_DIN3:
81  if ( di_cfg_request->polarity.din3 != di_cfg.polarity.din3 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
82  if ( di_cfg_request->swap_din1_and_din3 != di_cfg.swap_din1_and_din3 ) error_code = EC_STX104_SPI_DI_CFG_COLLISION_ERROR;
83  break;
84  }
85  if ( error_code ) goto STX104_SPI_DI_Config_Request_Check_Error;
86  return SUCCESS;
87 STX104_SPI_DI_Config_Request_Check_Error:
88  if ( error_code < 0 ) error_code = -error_code;
89  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
90  return -error_code;
91 }

187 AUL_DLL_API int AULCALLCONV STX104_SPI_CFG_Set( int board, STX104_SPI_CFG_TYPE * cfg )
188 {
189  int error_code;
190  devmgr_device_type * device;
191  uint16_t reg_uint16;
192  double time_max_ns;
193  double time_half_ns;
194  double time_ecd_ns;
195  /* SCHI[11:0] */
196  time_max_ns = STX104_CLOCK_RESOLUTION_NS * ( 4.0 + 4095.0 );
197  time_half_ns = ((double) cfg->half_clock_ns );
198  if ( time_half_ns > time_max_ns )
199  { /*AUL_Error_Log("STX104_SPI_CFG_Set: Error - half clock time too long\n" );*/
200  return EC_STX104_SPI_HALF_CLOCK_ERROR;
201  }
202  /* half_clock ECD[3:0] */
203  time_max_ns = STX104_CLOCK_RESOLUTION_NS * 4.0 + time_max_ns * 15.0;
204  time_ecd_ns = ((double) cfg->end_cycle_delay_ns );
205  if ( time_ecd_ns > time_max_ns )
206  { error_code = EC_STX104_SPI_END_OF_CYCLE_DELAY_ERROR; goto STX104_SPI_CFG_Set_Error; }
207  if ( ( cfg->sdi_map < SPI_SDI_DIN0 ) || ( cfg->sdi_map > SPI_SDI_DIN3 ) )
208  { error_code = EC_STX104_SPI_SDI_MAP_ERROR; goto STX104_SPI_CFG_Set_Error; }
209  if ( ( cfg->chip_select_behavior < SPI_CSB_SOFTWARE ) || ( cfg->chip_select_behavior > SPI_CSB_TOGGLE_16 ) )
210  { error_code = EC_STX104_SPI_CSB_ERROR; goto STX104_SPI_CFG_Set_Error; }
211 
212  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_SPI_CFG_Set_Error;
213  error_code = STX104_SPI_Firmware_Revision_Check( board, "STX104_SPI_CFG_Set" );
214  if ( error_code ) goto STX104_SPI_CFG_Set_Error;
215  /* half clock */
216  time_half_ns = ((double) cfg->half_clock_ns );
217  time_half_ns = ( time_half_ns / STX104_CLOCK_RESOLUTION_NS ) - 4.0;
218  reg_uint16 = (uint16_t) time_half_ns;
219  STX104_Index_Write_U16( board, STX104_SPI_HALF_CLOCK, reg_uint16 );
220  time_half_ns = ((double) cfg->half_clock_ns );
221  /* sdi_map */
222  reg_uint16 = ( (uint16_t) cfg->sdi_map ) & 0x0003;
223  /* compute End of Cycle Delay (ECD) */
224  time_ecd_ns = ((double) cfg->end_cycle_delay_ns );
225  time_ecd_ns = ( time_ecd_ns - ( STX104_CLOCK_RESOLUTION_NS * 4.0 ) ) / time_half_ns;
226  reg_uint16 = reg_uint16 << 4;
227  reg_uint16 |= ( ((uint16_t) time_ecd_ns ) & 0x000F );
228 
229  reg_uint16 = reg_uint16 << 1;
230  if ( cfg->sdio_wrap ) reg_uint16 |= 0x0001;
231 
232  reg_uint16 = reg_uint16 << 3;
233  reg_uint16 |= ( (uint16_t) cfg->chip_select_behavior ) & 0x0003;
234 
235  reg_uint16 = reg_uint16 << 1;
236  if ( cfg->sdo_polarity ) reg_uint16 |= 0x0001;
237 
238  reg_uint16 = reg_uint16 << 1;
239  if ( cfg->sdi_polarity ) reg_uint16 |= 0x0001;
240 
241  reg_uint16 = reg_uint16 << 1;
242  if ( cfg->sclk_phase ) reg_uint16 |= 0x0001;
243 
244  reg_uint16 = reg_uint16 << 1;
245  if ( cfg->sclk_polarity ) reg_uint16 |= 0x0001;
246 
247  STX104_Index_Write_U16( board, STX104_SPI_CONFIGURATION, reg_uint16 );
248  return SUCCESS;
249 STX104_SPI_CFG_Set_Error:
250  if ( error_code < 0 ) error_code = -error_code;
251  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
252  return -error_code;
253 }

256 AUL_DLL_API int AULCALLCONV STX104_SPI_CFG_Get( int board, STX104_SPI_CFG_TYPE * cfg )
257 {
258  int error_code;
259  devmgr_device_type * device;
260  uint16_t reg_uint16;
261  double time_ns;
262 
263  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_SPI_CFG_Get_Error;
264 
265  error_code = STX104_SPI_Firmware_Revision_Check( board, "STX104_SPI_CFG_Get" );
266  if ( error_code ) goto STX104_SPI_CFG_Get_Error;
267 
268  Mem_Set( cfg, 0, sizeof( STX104_SPI_CFG_TYPE ) );
269  /* SPI Configuration Register */
270  STX104_Index_Read_U16( board, STX104_SPI_CONFIGURATION, &reg_uint16 );
271 
272  cfg->sdi_map = (STX104_SPI_SDI_ENUM) ( ( reg_uint16 & 0x3000 ) >> 12 );
273 
274  /* NOTE: actual time in ns is calculated down at the bottom of this function */
275  cfg->end_cycle_delay_ns = (uint32_t) ( ( reg_uint16 & 0x0F00 ) >> 8 );
276 
277  if ( ( reg_uint16 & 0x0080 ) == 0x0080 ) cfg->sdio_wrap = true;
278 
279  cfg->chip_select_behavior = (STX104_SPI_CSB_ENUM) ( ( reg_uint16 & 0x0030 ) >> 4 );
280 
281  if ( ( reg_uint16 & 0x0008 ) == 0x0008 ) cfg->sdo_polarity = true;
282 
283  if ( ( reg_uint16 & 0x0004 ) == 0x0004 ) cfg->sdi_polarity = true;
284 
285  if ( ( reg_uint16 & 0x0002 ) == 0x0002 ) cfg->sclk_phase = true;
286 
287  if ( ( reg_uint16 & 0x0001 ) == 0x0001 ) cfg->sclk_polarity = true;
288 
289  /* half clock */
290  STX104_Index_Read_U16( board, STX104_SPI_HALF_CLOCK, &reg_uint16 );
291  reg_uint16 = reg_uint16 & 0x0FFF; /* mask so we only get lower 12 bits */
292  time_ns = STX104_CLOCK_RESOLUTION_NS * ( (double)( reg_uint16 + 4 ) );
293  cfg->half_clock_ns = (uint16_t) time_ns;
294  /* half-clock */
295  time_ns = ((double) cfg->end_cycle_delay_ns ) * time_ns + STX104_CLOCK_RESOLUTION_NS * 4.0;
296  cfg->end_cycle_delay_ns = (uint32_t) time_ns;
297  return SUCCESS;
298 STX104_SPI_CFG_Get_Error:
299  if ( error_code < 0 ) error_code = -error_code;
300  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
301  return -error_code;
302 }

406 AUL_DLL_API int AULCALLCONV STX104_SPI_Transmit( int board,
407  uint8_t * tx_buffer,
408  uint8_t tx_count,
409  uint8_t * tx_count_actual
410  )
411 {
412  int error_code;
413  devmgr_device_type * device;
414  uint8_t tx_index;
415  uint8_t tx_index_max;
416  BOOL not_done;
417 /* BOOL rx_not_empty;*/
418  uint8_t status;
419  uint8_t tx_byte;
420 /* size_t index;*/
421 /* uint8_t buffer[STX104_SPI_BUFFER_LENGTH_MAX];*/
422  (void) tx_count_actual;
423  tx_index = 0;
424 
425  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_SPI_Transmit_Error;
426  error_code = STX104_SPI_Active_Check( board, "STX104_SPI_Transceiver" );
427  if ( error_code ) goto STX104_SPI_Transmit_Error;
428 
429  if ( 0 == tx_count ) return( error_code );
430 
431  STX104_SPI_Chip_Select_Commit_Set( board, false ); /* remove commit */
432 
433  /* prefill the TX buffer */
434  not_done = true;
435  tx_index = 0;
436  tx_index_max = tx_count - 1;
437  while ( not_done )
438  {
439  STX104_Index_Read_U8( board, STX104_SPI_STATUS, &status );
440 
441  if ( !( status & STX104_SPI_TX_FULL ) )
442  {
443  tx_byte = tx_buffer[tx_index];
444  STX104_Index_Write_U8( board, STX104_SPI_TRANSMIT_FIFO_DATA, tx_byte );
445  if ( tx_index < tx_index_max ) tx_index++;
446  else not_done = false; /* done writing to buffer */
447  }
448  else
449  {
450 /* AUL_Error_Log("STX104_SPI_Transceiver: Error - TX Full Error.\n" );*/
451  error_code = EC_STX104_SPI_TX_FULL;
452  not_done = false;
453  break;
454  }
455  }
456 
457  if ( error_code ) goto STX104_SPI_Transmit_Error;
458 
459  STX104_SPI_Chip_Select_Commit_Set( board, true ); /* commit the TX buffer */
460  return SUCCESS;
461 STX104_SPI_Transmit_Error:
462  if ( error_code < 0 ) error_code = -error_code;
463  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
464  return -error_code;
465 }
