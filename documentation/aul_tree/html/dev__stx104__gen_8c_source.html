52 \param[out] jumpers A pointer to the ::STX104_GEN_JMP_TYPE structure that will be used to retrieve the jumper settings.
53 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
54 */
55 AUL_DLL_API int AULCALLCONV STX104_Gen_Jumpers_Get( int board, STX104_GEN_JMP_TYPE * jumpers_status )
56 {
57  int error_code;
58  devmgr_device_type * device;
59  size_t address;
60  uint8_t gen_reg;
61  uint8_t adc_stat_reg;
62  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_Gen_Revision_Get_Error;
63 
64  Mem_Set( jumpers_status, 0, sizeof( STX104_GEN_JMP_TYPE ) );
65 
66  STX104_Index_Read_U8( board, STX104_GENERAL_CONFIGURATION, &gen_reg );
67  if ( ( gen_reg & 0x01 ) == 0x01 ) jumpers_status->M0_mode = true;
68  if ( ( gen_reg & 0x02 ) == 0x02 ) jumpers_status->M1_mode = true;
69  if ( ( gen_reg & 0x04 ) == 0x04 ) jumpers_status->M2_dacs_16bits = true;
70  if ( ( gen_reg & 0x08 ) == 0x08 ) jumpers_status->M3_moving_average_filter = true;
71  if ( ( gen_reg & 0x10 ) == 0x10 ) jumpers_status->M4_address_10bit_mode = true;
72  if ( ( gen_reg & 0x20 ) == 0x20 ) jumpers_status->DMA_ch3 = true;
73  if ( ( gen_reg & 0x40 ) == 0x40 ) jumpers_status->CLK_1mhz = true;
74 
75  address = device->io_region_offset + STX104_ADC_STATUS;
76  IO_Read_U8( device->io_device, address, &adc_stat_reg );
77  if ( ( adc_stat_reg & 0x40 ) == 0x40 ) jumpers_status->J9_ai_unipolar = true;
78  if ( ( adc_stat_reg & 0x20 ) == 0x20 ) jumpers_status->J8_ai_differential = true;
79 
80  return error_code;
81 STX104_Gen_Revision_Get_Error:
82  if ( error_code < 0 ) error_code = -error_code;
83  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
84  return -error_code;
85 }
