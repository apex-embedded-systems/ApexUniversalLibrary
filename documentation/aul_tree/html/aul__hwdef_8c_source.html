136 static void AULHwDef_String_Trim_Left( char* string )
137 {
138  int src;
139  int dst;
140  BOOL non_space_copy;
141 
142  non_space_copy = false;
143  src = 0;
144  dst = 0;
145  while( string[src] )
146  {
147  if ( non_space_copy )
148  {
149  string[dst] = string[src];
150  dst++;
151  }
152  else if ( ( string[src] != ' ' ) && ( string[src] != '\t' /* tab */ ) )
153  {
154  string[dst] = string[src];
155  dst++;
156  non_space_copy = true;
157  }
158  src++;
159  }

334 static void AULHwDef_Split_File_Name( const char * name,
335  char * directory,
336  int dir_length,
337  char * file_name,
338  int file_name_length
339  )
340 {
341  int length;
342  int cp;
343  int cpd;
344  int dir_last_char;
345 
346  length = String_Length( name );
347  dir_last_char = -1;
348  for( cp=length-1; cp > -1; cp-- )
349  {
350  if ( (name[cp] == '\\') || (name[cp] == '/') || (name[cp] == ':') )
351  {
352  dir_last_char = cp;
353  cp = -1;
354  }
355  }
356  /* directory portion of string */
357  cpd = 0;
358  if ( dir_last_char >= 0 )
359  {
360  cpd = dir_last_char + 1;
361  if ( cpd >= (dir_length - 1) )
362  { /* not enough room to fit the directory string - return a null string */
363  cpd = 0;
364  }
365  else
366  {
367  String_Copy( directory, name, cpd );
368  }
369  }
370  directory[cpd] = '\0';
371  /* filename portion of string */
372  cp = 0;
373  if ( length >= 0 )
374  {
375  cp = length - cpd;
376  if ( cp > (file_name_length - 1) )
377  { /* not enough room to fit the filename string - return a null string */
378  cp = 0;
379  }
380  else
381  {
382  String_Copy( file_name, name + cpd, cp );
383  }
384  }
385  file_name[cp] = '\0';
386 }

388 */
389 char * String_Tokenize( char * str, char ** token_state )
390 {
391  char * cp;
392 
393  if ( NULL == str )
394  { /* return next token, if any */
395  str = *token_state;
396  if ( NULL == str ) return NULL; /* no more tokens */
397  }
398  else
399  { /* find the start of current token */
400  cp = str;
401  while ( ( *cp == ',' ) && ( *cp != '\0' ) ) cp++;
402  if ( *cp != '\0' ) str = cp;
403  }
404  /* find the next token position and put null character there */
405  cp = str;
406  while ( ( *cp != ',' ) && ( *cp != '\0' ) ) cp++;
407  if ( cp != '\0' )
408  { /* found another delimiter */
409  *cp = '\0';
410  *token_state = cp + 1; /* next token position */
411  }
412  else
413  { /* no more tokens */
414  *token_state = NULL;
415  }
416  return str;
417 }
