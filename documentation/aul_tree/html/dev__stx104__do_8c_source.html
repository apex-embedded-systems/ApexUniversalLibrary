104 /**********************************************************************************************************//**
105 @ingroup stx104_do
106 \brief
107 Apex Univeral Library STX104 specific function that retireves the digital output (DO) configuration.
108 
109 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
110 \param[out] cfg A pointer to the ::stx104_do_cfg_type structure that will be used to retrieve the digital output configuration information.
111 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
112 */
113 AUL_DLL_API int AULCALLCONV STX104_DO_CFG_Get( int board, stx104_do_cfg_type * cfg )
114 {
115  int error_code;
116  devmgr_device_type * device;
117  uint8_t reg_value_low;
118  uint16_t reg_value;
119  uint16_t firmware_revision;
120  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_DO_CFG_Get_Error;
121 
122  Mem_Set( cfg, 0, sizeof( stx104_do_cfg_type ) );
123  STX104_Gen_Revision_Get( board, &firmware_revision );
124  if ( firmware_revision >= STX104_REVISION_140101 )
125  { /* get the extended digital output configuration */
126  STX104_Index_Read_U16( board, STX104_DIGITAL_OUTPUT_CONFIGURATION, &reg_value );
127  reg_value_low = (uint8_t) ( reg_value & 0x00FF );
128 
129  reg_value = reg_value >> 8;
130  cfg->do0_map = (STX104_DO0_MAP_ENUM)(reg_value & 0x03 );
131  reg_value = reg_value >> 2;
132  cfg->do1_map = (STX104_DO1_MAP_ENUM)(reg_value & 0x03 );
133  reg_value = reg_value >> 2;
134  cfg->do2_map = (STX104_DO2_MAP_ENUM)(reg_value & 0x03 );
135  reg_value = reg_value >> 2;
136  cfg->do3_map = (STX104_DO3_MAP_ENUM)(reg_value & 0x03 );
137  }
138  else
139  {
140  STX104_Index_Read_U8( board, STX104_DIGITAL_OUTPUT_CONFIGURATION, &reg_value_low );
141  }
142  /* convert lower bits */
143  if ( reg_value_low & 0x80 ) cfg->polarity.sshp = true;
144  if ( reg_value_low & 0x20 ) cfg->polarity.ct_out0 = true;
145  if ( reg_value_low & 0x10 ) cfg->polarity.ct_out2 = true;
146  if ( reg_value_low & 0x08 ) cfg->polarity.dout3 = true;
147  if ( reg_value_low & 0x04 ) cfg->polarity.dout2 = true;
148  if ( reg_value_low & 0x02 ) cfg->polarity.dout1 = true;
149  if ( reg_value_low & 0x01 ) cfg->polarity.dout0 = true;
150 
151  return error_code;
152 STX104_DO_CFG_Get_Error:
153  if ( error_code < 0 ) error_code = -error_code;
154  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
155  return -error_code;
156 }
157 /**********************************************************************************************************//**
158 @ingroup stx104_do
159 \brief
160 Apex Univeral Library STX104 specific function that sets the digital output (DO) configuration.
161 
162 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
163 \param[in] cfg A pointer to the ::stx104_do_cfg_type structure that will be used to set the digital output configuration information.
164 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
165 */
166 AUL_DLL_API int AULCALLCONV STX104_DO_CFG_Set( int board, stx104_do_cfg_type * cfg )
167 {
168  int error_code;
169  devmgr_device_type * device;
170  uint8_t reg_value_low;
171  uint16_t reg_value;
172  uint16_t firmware_revision;
173  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_DO_CFG_Set_Error;
174 #if defined( STX104_SPI_H_ )
175  /* test to make sure that we are not interferring with active SPI interface */
176  error_code = STX104_SPI_DO_Config_Request_Check( board, cfg, "STX104_DO_CFG_Set" );
177  if ( error_code ) goto STX104_DI_CFG_Set_Error;
178 #endif
179  reg_value_low = 0;
180  if ( cfg->polarity.sshp ) reg_value_low |= 0x80;
181  if ( cfg->polarity.ct_out0 ) reg_value_low |= 0x20;
182  if ( cfg->polarity.ct_out2 ) reg_value_low |= 0x10;
183  if ( cfg->polarity.dout3 ) reg_value_low |= 0x08;
184  if ( cfg->polarity.dout2 ) reg_value_low |= 0x04;
185  if ( cfg->polarity.dout1 ) reg_value_low |= 0x02;
186  if ( cfg->polarity.dout0 ) reg_value_low |= 0x01;
187 
188  STX104_Gen_Revision_Get( board, &firmware_revision );
189  if ( firmware_revision >= STX104_REVISION_140101 )
190  { /* update the extended digital output configuration */
191  reg_value = ((uint16_t) cfg->do3_map) & 0x03;
192  reg_value = reg_value << 2;
193  reg_value |= ((uint16_t) cfg->do2_map) & 0x03;
194  reg_value = reg_value << 2;
195  reg_value |= ((uint16_t) cfg->do1_map) & 0x03;
196  reg_value = reg_value << 2;
197  reg_value |= ((uint16_t) cfg->do0_map) & 0x03;
198  reg_value = reg_value << 8;
199  reg_value |= ((uint16_t) reg_value_low ) & 0xFF;
200  STX104_Index_Write_U16( board, STX104_DIGITAL_OUTPUT_CONFIGURATION, reg_value );
201  }
202  else
203  {
204  STX104_Index_Write_U8( board, STX104_DIGITAL_OUTPUT_CONFIGURATION, reg_value_low );
205  }
206  return error_code;
207 STX104_DO_CFG_Set_Error:
208  if ( error_code < 0 ) error_code = -error_code;
209  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
210  return -error_code;
211 }

216 Apex Univeral Library STX104 specific function that retrieves the specified digital output (DO) channel.
217 
218 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
219 \param[in] channel The channel number that the data value will be read from; STX104_DO0 (0) <= channel <= STX104_DO3 (3) (i.e. ::STX104_DO_ENUM).
220 \param[out] value Pointer to the variable that will receive the digital output value. This is a copy of the last value written.
221 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
222 */
223 AUL_DLL_API int AULCALLCONV STX104_DO_Get( int board, STX104_DO_ENUM channel, BOOL * value )
224 {
225  int error_code;
226  devmgr_device_type * device;
227  uint8_t mask;
228  stx104_dataset_type * ds;
229  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_DO_Get_Error;
230  ds = (stx104_dataset_type *) device->private_data;
231  if ( (channel >= STX104_DO0) && (channel <= STX104_DO3) )
232  {
233  mask = 0x01 << channel;
234  if ( ds->do_reg & mask ) *value = true;
235  else *value = false;
236  }
237  else
238  {
239  error_code = EC_STX104_CHANNEL_OUT_OF_RANGE; goto STX104_DO_Get_Error;
240  }
241  return error_code;
242 STX104_DO_Get_Error:
243  if ( error_code < 0 ) error_code = -error_code;
244  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
245  return -error_code;

258 AUL_DLL_API int AULCALLCONV STX104_DO_Set( int board, STX104_DO_ENUM channel, BOOL value )
259 {
260  int error_code;
261  devmgr_device_type * device;
262  size_t address;
263  uint8_t mask;
264  stx104_dataset_type * ds;
265  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_DO_Set_Error;
266  ds = (stx104_dataset_type *) device->private_data;
267 #if defined( STX104_SPI_H_ )
268  /* test to make sure that we are not interferring with active SPI interface */
269  error_code = STX104_SPI_DO_Set_Request_Check( board, channel, "STX104_DO_Set" );
270  if ( error_code ) goto STX104_DO_Set_Error;
271 #endif
272 
273  if ( (channel >= STX104_DO0) && (channel <= STX104_DO3) )
274  {
275  mask = 0x01 << channel;
276 
277  if ( value == false ) ds->do_reg &= ~mask;
278  else ds->do_reg |= mask;
279 
280  address = device->io_region_offset + STX104_DIGITAL_OUTPUTS;
281  IO_Write_U8( device->io_device, address, ds->do_reg );
282  }

286  }
