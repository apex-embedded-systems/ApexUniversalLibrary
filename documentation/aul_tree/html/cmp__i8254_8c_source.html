175 static void I8254_CT_Xfr_Read_Helper( component_type * component,
176  I8254_CHANNEL_ENUM channel,
177  uint8_t control,
178  uint8_t * status,
179  uint16_t * count
180  )
181 {
182  I8254_RW_ENUM rw;
183  uint8_t u8_lo = 0;
184  uint8_t u8_hi = 0;
185  /* read the status */
186  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &u8_lo, sizeof( u8_lo ) );
187  /* mask off the output and null count bits */
188  u8_lo = u8_lo & ~( I8254_CTRL_MASK_SC << I8254_CTRL_SHIFT_SC );
189  /* now, insert the channel number - yes we are ignoring output bit and null count bit. */
190  u8_lo = u8_lo | I8254_CTRL_SET_SC( channel );
191  rw = I8254_CTRL_GET_RW( u8_lo );
192  *status = u8_lo;
193  /* if there is no count readback then we are done */
194  if ( control & ~I8254_READBACK_LATCH_COUNT ) return;
195  switch ( rw )
196  {
197  case I8254_RW_LSB:
198  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
199  break;
200  case I8254_RW_MSB:
201  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
202  break;
203  case I8254_RW_LSB_MSB:
204  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
205  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
206  break;
207  default:
208  break;
209  }
210  *count = IO_U16_SET_U8_LO_HI(u8_lo,u8_hi);
211 }

213 static int I8254_CMD_Xfr( component_type * component,
214  size_t index, /* not used */
215  char * buf,
216  size_t buf_len /* not used */
217  )
218 {
219  int error_code;
220  I8254_CHANNEL_ENUM channel;
221  I8254_RW_ENUM rw;
222  /*i8254_dataset_type * ds;*/
223  i8254_ct_xfr_type cxfr;
224  uint8_t control;
225  (void) buf_len;
226  (void) index;
227 
228  if ( NULL == buf )
229  { error_code = EC_CMP_BUF_NULL; goto I8254_CT_Xfr_Done; }
230 
231  /* get transfer command from user space */
232  if ( IO_COPY_FROM_USER( &cxfr, buf, sizeof( cxfr ) ) )
233  { error_code = EC_IO_COPY_FROM_USER; goto I8254_CT_Xfr_Done; }
234 
235  if ( SYS_TYPE_MAKE( i8254_ct_xfr_type ) != cxfr.type )
236  { error_code = EC_CMP_TYPE_TYPES_DO_NOT_MATCH; goto I8254_CT_Xfr_Done; }
237 
238  /*ds = ( i8254_dataset_type *) component->private_data;*/
239 
240  /* Write the control value to the I8254 control register */
241  control = cxfr.control;
242  channel = I8254_CTRL_GET_SC(control);
243  rw = I8254_CTRL_GET_RW(control);
244  error_code = SUCCESS;
245  if ( I8254_CHANNEL_READBACK == channel )
246  { /* Read count value. Note: CT offset address just happens to match channel #. */
247  int chan_bits = I8254_CTRL_GET_MODE(control);
248  /* if neither count or status latch is low, then nothing to do */
249  if ( I8254_READBACK_LATCH_NONE == ( control & I8254_READBACK_LATCH_NONE ) ) goto I8254_CT_Xfr_Done;
250 
251  /* Need to readback the status as well to know how to deal with read/write. */
252  control = cxfr.control & I8254_READBACK_LATCH_STATUS;
253  CmpMgr_Reg_Write( component, I8254_CONTROL, (char *) &( control ), sizeof( control ) );
254 
255  /* process one or more channels as dictated by the chan_bits from the control word */
256  if ( I8254_MODE_READBACK_CHAN_0 & chan_bits )
257  {
258  channel = I8254_CHANNEL_0;
259  I8254_CT_Xfr_Read_Helper( component, channel, cxfr.control,
260  &(cxfr.status_read[channel]), &(cxfr.count_read[channel]) );
261  }
262  if ( I8254_MODE_READBACK_CHAN_1 & chan_bits )
263  {
264  channel = I8254_CHANNEL_1;
265  I8254_CT_Xfr_Read_Helper( component, channel, cxfr.control,
266  &(cxfr.status_read[channel]), &(cxfr.count_read[channel]) );
267  }
268  if ( I8254_MODE_READBACK_CHAN_2 & chan_bits )
269  {
270  channel = I8254_CHANNEL_2;
271  I8254_CT_Xfr_Read_Helper( component, channel, cxfr.control,
272  &(cxfr.status_read[channel]), &(cxfr.count_read[channel]) );
273  }
274  }
275  else if ( I8254_RW_LATCH == rw )
276  { /* latch a single channel and read out the count value */
277  I8254_MODE_ENUM mode;
278  uint8_t u8_lo = 0;
279  uint8_t u8_hi = 0;
280  switch ( channel )
281  {
282  case I8254_CHANNEL_0: mode = I8254_MODE_READBACK_CHAN_0; break;
283  case I8254_CHANNEL_1: mode = I8254_MODE_READBACK_CHAN_1; break;
284  case I8254_CHANNEL_2: mode = I8254_MODE_READBACK_CHAN_2; break;
285  default:
286  error_code = EC_I8254_CHANNEL_OUT_OF_RANGE;
287  goto I8254_CT_Xfr_Done; /* wrong channel info */
288  break;
289  }
290  /* Need to readback status of the counter to know how to read out the count */
291  I8254_Control_Create( I8254_CHANNEL_READBACK,
292  I8254_RW_STA,
293  mode, /* readback channel for status */
294  I8254_TYPE_BINARY, /* readback ignores this */
295  &control
296  );
297  /* write the command to capture status and then read it. */
298  CmpMgr_Reg_Write( component, I8254_CONTROL, (char *) &( control ), sizeof( control ) );
299  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &(cxfr.status_read[channel]), sizeof( cxfr.status_read[channel] ) );
300  rw = I8254_CTRL_GET_RW( cxfr.status_read[channel] );
301  /* now write the command to actually latch the data */
302  CmpMgr_Reg_Write( component, I8254_CONTROL, (char *) &( cxfr.control ), sizeof( cxfr.control ) );
303  switch ( rw )
304  {
305  case I8254_RW_LSB:
306  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
307  break;
308  case I8254_RW_MSB:
309  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
310  break;
311  case I8254_RW_LSB_MSB:
312  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
313  CmpMgr_Reg_Read( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
314  break;
315  default:
316  break;
317  }
318  cxfr.count_read[channel] = IO_U16_SET_U8_LO_HI(u8_lo,u8_hi); /* save the count value to the counter transfer */
319  }
320  else
321  { /* Write count value */
322  uint8_t u8_lo = IO_U16_GET_U8_LO( cxfr.count_write );
323  uint8_t u8_hi = IO_U16_GET_U8_HI( cxfr.count_write );
324 
325  CmpMgr_Reg_Write( component, I8254_CONTROL, (char *) &( control ), sizeof( control ) );
326  /* Write count value. Note: CT offset address just happens to match channel # */
327  switch ( rw )
328  {
329  case I8254_RW_LSB:
330  CmpMgr_Reg_Write( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
331  break;
332  case I8254_RW_MSB:
333  CmpMgr_Reg_Write( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
334  break;
335  case I8254_RW_LSB_MSB:
336  CmpMgr_Reg_Write( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_lo ), sizeof( u8_lo ) );
337  CmpMgr_Reg_Write( component, channel /* I8254_CHANNEL_n */, (char *) &( u8_hi ), sizeof( u8_hi ) );
338  break;
339  default:
340  break;
341  }
342  /* mask off channel info - for now ignore what D7 and D6 might be. */
343  cxfr.status_read[channel] = cxfr.control & ~( I8254_CTRL_MASK_SC << I8254_CTRL_SHIFT_SC );
344  }
345 I8254_CT_Xfr_Done:
346  cxfr.error_code = error_code;
347  /* copy results back to user space */
348  if ( IO_COPY_TO_USER( buf, &cxfr, sizeof( cxfr ) ) ) error_code = EC_IO_COPY_TO_USER;
349  if ( SUCCESS == error_code ) return error_code;
350  if ( error_code < 0 ) error_code = -error_code;
351  Error_Log( COMPONENT_I8254_NAME, __func__, __LINE__, error_code );
352  return -error_code;
353 }

408 static int I8254_Restart( component_type * component,
409  size_t index,
410  char * buf,
411  size_t buf_len
412  )
413 {
414  int error_code;
415  size_t operation;
416  i8254_dataset_type * ds;
417  (void) index;
418  (void) buf_len;
419 
420  if ( NULL == buf ) operation = RESTART_COLD;
421  else if ( 0 == *buf ) operation = RESTART_COLD;
422  else operation = RESTART_WARM;
423 
424  error_code = SUCCESS;
425  ds = ( i8254_dataset_type *) component->private_data;
426  switch( operation )
427  { /*--------------------------------*/
428  case RESTART_COLD:
429  { /* commit all configuration to hardware */
430  i8254_ct_xfr_array_type cxa;
431  cxa.type = SYS_TYPE_MAKE( i8254_ct_xfr_array_type );
432  cxa.qty = I8254_CT_QTY;
433  cxa.cxfr = ds->cfg.cxfr; /* use configuration array */
434  error_code = I8254_CMD_Xfr_Array( component, 0, (char *) &cxa, 0 );
435  }
436  break;
437  /*--------------------------------*/
438  case RESTART_WARM:
439  { /* retrieve all that we can from the hardware */
440  i8254_ct_xfr_type cxfr;
441  i8254_ct_xfr_type * cfg_cxfr;
442  size_t chan;
443 
444  Mem_Set( &cxfr, 0, sizeof( cxfr ) );
445  cxfr.control = I8254_CTRL_CREATE( I8254_CHANNEL_READBACK, \
446  I8254_RW_CNT_STA, \
447  I8254_MODE_READBACK_CHAN_ALL, \
448  I8254_TYPE_BINARY /* ignored */ \
449  );
450  error_code = I8254_CMD_Xfr( component, 0, (char *) &cxfr, 0 );
451  cxfr.error_code = error_code;
452  cfg_cxfr = ds->cfg.cxfr;
453 
454  for ( chan = 0; chan < I8254_CT_QTY; chan++ )
455  {
456  cfg_cxfr[chan].error_code = cxfr.error_code;
457  cfg_cxfr[chan].type = SYS_TYPE_MAKE( i8254_ct_xfr_type );
458  cfg_cxfr[chan].status_read[0] = cxfr.status_read[chan]; /* raw status */
459  /* mask off channel info */
460  cfg_cxfr[chan].control = cxfr.status_read[chan] & ~( I8254_CTRL_MASK_SC << I8254_CTRL_SHIFT_SC );
461  /* insert channel info -- this completes reconstruction of the control value */
462  cfg_cxfr[chan].control |= I8254_CTRL_SET_SC( chan );
463  cfg_cxfr[chan].count_read[0] = cxfr.count_read[chan];
464  }
465  }
466  break;
467  /*--------------------------------*/
468  default: /* cold or any other */
469 /* copy intialize values from config into hardware registers. more like a reset.*/
470  break;
471  }
472  return error_code;
473 }

630 \param[in] ct_clock_source
631 \param[in] channel
632 \param[in] control
633 \param[out] initial_count

635 */
636 AUL_DLL_API int AULCALLCONV I8254_Count_Freq_Hz( double requested_hz, /* requested freq */
637  double clock_source_hz, /* default, used if not CTn_nMHz */
638  int ct_clock_source, /* TRACERE_CLK_CTn_4MHZ _10MHZ _20MHZ _40MHZ */
639  uint8_t control, /* i8254 control word */
640  uint16_t * initial_count, /* i8254 count to load */
641  double * actual_hz, /* actual output frequency */
642  double * error_actual /* computed error, x1.0e6 to get ppm */
643  )
644 {
645  double clk_freq;
646  double divisor;
647  double scratch;
648  int error_code;
649  I8254_MODE_ENUM mode;
650  uint16_t divisor_u16;
651 
652  clk_freq = clock_source_hz;
653 
654  if ( ( 0.0 == clk_freq ) || ( clk_freq > 40.0e6 ) )
655  { error_code = EC_I8254_CLK_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
656 
657  if ( ( requested_hz > clk_freq ) || ( 0.0 == requested_hz ) )
658  { error_code = EC_I8254_FREQ_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
659 
660  mode = I8254_CTRL_GET_MODE( control );
661  switch( mode )
662  {
663  case I8254_MODE_RATE_GENERATOR:
664  divisor = clk_freq / requested_hz;
665  break;
666  case I8254_MODE_SQUARE_WAVE_GENERATOR:
667  if ( requested_hz > ( clk_freq / 2.0 ) )
668  { error_code = EC_I8254_FREQ_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
669  divisor = clk_freq / requested_hz;
670  break;
671  default:
672  error_code = EC_I8254_INVALID_MODE;
673  goto I8254_Count_Freq_Hz_Error;
674  }
675  if ( divisor > ((double) UINT16_MAX ) )
676  { error_code = EC_I8254_FREQ_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
677 
678  if ( divisor < 1.0 )
679  { error_code = EC_I8254_FREQ_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
680 
681  divisor_u16 = (uint16_t) divisor;
682  scratch = divisor - ( (double) divisor_u16 );
683  if ( scratch > 0.5 ) divisor_u16++;
684  else if ( scratch < -0.5 ) divisor_u16--;
685 
686  if ( 0 == divisor_u16 )
687  { error_code = EC_I8254_FREQ_OUT_OF_RANGE; goto I8254_Count_Freq_Hz_Error; }
688 
689  scratch = ( (double) divisor_u16 );
690  *actual_hz = clk_freq / scratch;
691  scratch = divisor - scratch;
692  *error_actual = scratch / divisor;
693  *initial_count = divisor_u16;
694  return SUCCESS;
695 I8254_Count_Freq_Hz_Error:
696  if ( error_code < 0 ) error_code = -error_code;
697  Error_Log( COMPONENT_I8254_NAME, __func__, __LINE__, error_code );
698  return -error_code;

703  Apex Univeral Library I8254 specific function
704

706 \param[in] clock_source_hz
707 \param[in] channel
708 \param[in] control

712 AUL_DLL_API int AULCALLCONV I8254_Count_Time_Sec( double requested_sec, /* requested freq */
713  double clock_source_hz, /* default, used if not CTn_nMHz */
714  uint8_t control, /* i8254 control word */
715  uint16_t * initial_count, /* i8254 count to load */
716  double * actual_sec, /* actual output time interval */
717  double * error_actual /* computed error, x1.0e6 to get ppm */
718  )
719 {
720  double clk_freq;
721  double interval_clock;
722  double count;
723  double scratch;
724  int error_code;
725  uint16_t count_u16;
726  I8254_MODE_ENUM mode;
727 
728  clk_freq = clock_source_hz;
729 
730  if ( ( 0.0 == clk_freq ) || ( clk_freq > 40.0e6 ) )
731  { error_code = EC_I8254_CLK_OUT_OF_RANGE; goto I8254_Count_Time_Sec_Error; }
732 
733  interval_clock = 1.0 / clk_freq; /* convert to frequency */
734  mode = I8254_CTRL_GET_MODE( control );
735  switch( mode )
736  {
737  case I8254_MODE_INTERRUPT_ON_TERMINAL_COUNT: /* (N + 1) * CLK */
738  count = ( requested_sec - interval_clock ) / interval_clock;
739  /* count must be zero or greater up to UINT16_MAX */
740  break;
741  case I8254_MODE_RATE_GENERATOR: /* N * CLK */
742  count = requested_sec / interval_clock;
743  break;
744  case I8254_MODE_SQUARE_WAVE_GENERATOR: /* N * CLK */
745  count = requested_sec / interval_clock;
746  break;
747  case I8254_MODE_HARDWARE_RETRIGGERABLE_ONE_SHOT: /* N * CLK */
748  count = requested_sec / interval_clock;
749  /* count must be zero or greater up to UINT16_MAX */
750  break;
751  case I8254_MODE_SOFTWARE_TRIGGERED_STROBE: /* (N + 1) * CLK */
752  count = ( requested_sec - interval_clock ) / interval_clock;
753  /* count must be zero or greater up to UINT16_MAX */
754  break;
755  case I8254_MODE_HARDWARE_TRIGGER_STROBE: /* (N + 1) * CLK */
756  count = ( requested_sec - interval_clock ) / interval_clock;
757  /* count must be zero or greater up to UINT16_MAX */
758  break;
759  default:
760  error_code = EC_I8254_INVALID_MODE;
761  goto I8254_Count_Time_Sec_Error;
762  }
763 
764  if ( ( count < 0.5 ) || ( count > ( ((double) UINT16_MAX) + 0.5 ) ) )
765  { error_code = EC_I8254_TIME_OUT_OF_RANGE; goto I8254_Count_Time_Sec_Error; }
766 
767  if ( ( count >= 0.5 ) && ( count < 1.0 ) ) count_u16 = 1;
768  else count_u16 = (uint16_t) count;
769 
770  if ( 0 == count_u16 ) count_u16 = UINT16_MAX;
771 
772  scratch = ( (double) count_u16 );
773  switch( mode )
774  {
775  case I8254_MODE_INTERRUPT_ON_TERMINAL_COUNT: /* (N + 1) * CLK */
776  *actual_sec = ( scratch + 1.0 ) * interval_clock;
777  break;
778  case I8254_MODE_RATE_GENERATOR: /* N * CLK */
779  *actual_sec = scratch * interval_clock;
780  break;
781  case I8254_MODE_SQUARE_WAVE_GENERATOR: /* N * CLK */
782  *actual_sec = scratch * interval_clock;
783  break;
784  case I8254_MODE_HARDWARE_RETRIGGERABLE_ONE_SHOT: /* N * CLK */
785  *actual_sec = scratch * interval_clock;
786  break;
787  case I8254_MODE_SOFTWARE_TRIGGERED_STROBE: /* (N + 1) * CLK */
788  *actual_sec = ( scratch + 1.0 ) * interval_clock;
789  break;
790  case I8254_MODE_HARDWARE_TRIGGER_STROBE: /* (N + 1) * CLK */
791  *actual_sec = ( scratch + 1.0 ) * interval_clock;
792  break;
793  default:
794  error_code = EC_I8254_INVALID_MODE;
795  goto I8254_Count_Time_Sec_Error;
796 
797  }
798  scratch = count - scratch;
799  *error_actual = scratch / count;
800  *initial_count = count_u16;
801  return SUCCESS;
802 I8254_Count_Time_Sec_Error:
803  if ( error_code < 0 ) error_code = -error_code;
804  Error_Log( COMPONENT_I8254_NAME, __func__, __LINE__, error_code );
805  return -error_code;
806 }
