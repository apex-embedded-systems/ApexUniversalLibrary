49 Apex Univeral Library STX104 specific function that retrieves the trigger active state.
50 
51 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
52 \param[out] state Pointer to the variable that will receive the trigger active state.
53 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.

55 AUL_DLL_API int AULCALLCONV STX104_Trg_Active_Get( int board, BOOL * state )
56 {
57  int error_code;
58  devmgr_device_type * device;
59  size_t address;
60  uint8_t value;
61  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_Trg_Active_Get_Error;
62  /*! get the ASR.TAS bit */
63  address = device->io_region_offset + STX104_ADC_CONFIGURATION;
64  IO_Read_U8( device->io_device, address, &value );
65  if ( (value & 0x08) == 0x08 ) *state = true;
66  else *state = false;

74 /**********************************************************************************************************//**
75 @ingroup stx104_trg
76 \brief
77 Apex Univeral Library STX104 specific function that retrieves the triggering configuration.
78 
79 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
80 \param[out] cfg A pointer to the ::STX104_TRG_CFG_TYPE structure that will be used to retrieve the analog input triggering configuration information.
81 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
82 */
83 AUL_DLL_API int AULCALLCONV STX104_Trg_CFG_Get( int board, STX104_TRG_CFG_TYPE * cfg )
84 {
85  int error_code;
86  devmgr_device_type * device;
87  uint16_t reg_value;
88  uint32_t tsd_reg;
89  double delay_ns;
90 
91  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_Trg_CFG_Get_Error;
92 
93  Mem_Set( cfg, 0, sizeof( STX104_TRG_CFG_TYPE ) );
94  STX104_Index_Read_U32( board, STX104_TRIGGER_START_DELAY, &tsd_reg );
95  delay_ns = ( (double) tsd_reg ) * STX104_CLOCK_RESOLUTION_NS;
96  cfg->delay_sec = delay_ns * 1e-9; /* convert to seconds */
97  cfg->delay_ns = (uint32_t) delay_ns;
98 
99  STX104_Index_Read_U16( board, STX104_TRIGGER_CONFIGURATION, &reg_value );
100 
101  if ( (reg_value & 0x8000) == 0x8000 ) cfg->enabled = true;
102 
103  cfg->start_source = (STX104_TRG_START_ENUM) ( reg_value & 0x000F );
104  cfg->stop_source = (STX104_TRG_STOP_ENUM) ( (reg_value & 0x00F0) >> 4 );
105  cfg->start_sync_source = (STX104_TRG_START_SYNC_ENUM)( (reg_value & 0x0F00) >> 8 );
106  cfg->start_mode = (STX104_TRG_MODE_ENUM) ( (reg_value & 0x3000) >> 12 );
107  return error_code;
108 STX104_Trg_CFG_Get_Error:
109  if ( error_code < 0 ) error_code = -error_code;
110  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
111  return -error_code;
112 }
113 /**********************************************************************************************************//**
114 @ingroup stx104_trg
115 \brief
116 Apex Univeral Library STX104 specific function that sets the triggering configuration.
117 
118 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
119 \param[in] cfg A pointer to the STX104_TRG_CFG_TYPE structure that will be used to set the analog input triggering configuration information.
120 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
121 */
122 AUL_DLL_API int AULCALLCONV STX104_Trg_CFG_Set( int board, STX104_TRG_CFG_TYPE * cfg )
123 {
124  int error_code;
125  devmgr_device_type * device;
126  uint16_t reg_value;
127  uint32_t tsd_reg;
128  double delay;
129  double delay_max_sec;
130  double delay_sec;
131 
132  if ( SUCCESS != ( error_code = AUL_Board( board, &device ) ) ) goto STX104_Trg_CFG_Set_Error;
133 
134  if ( cfg->enabled != 0 ) reg_value = 0x8000;
135  else reg_value = 0x0000;
136 
137  reg_value |= (uint16_t)( (cfg->start_mode & 0x03) << 12 );
138  reg_value |= (uint16_t)( (cfg->start_sync_source & 0x0F) << 8 );
139  reg_value |= (uint16_t)( (cfg->stop_source & 0x0F) << 4 );
140  reg_value |= (uint16_t)( (cfg->start_source & 0x0F) );
141 
142  delay_max_sec = STX104_CLOCK_RESOLUTION_NS * ((double) UINT32_MAX);
143  delay_max_sec = delay_max_sec * 1.0e-9; /* convert to seconds */
144 
145  /* delay computation */
146  delay_sec = 0.0;
147  if ( cfg->delay_ns > 0 )
148  {
149  delay_sec = ((double) cfg->delay_ns ) * 1.0e-9; /* convert to seconds */
150  }
151  if ( cfg->delay_sec > 0 )
152  {
153  delay_sec = cfg->delay_sec;
154  }
155 
156  /* cap the value */
157  if ( delay_sec > delay_max_sec )
158  {
159  delay = delay_max_sec;
160  }
161  else
162  {
163  delay = delay_sec;
164  }
165 
166  /* convert to nS */
167  delay = delay * 1.0e9;
168  /* convert to clock ticks */
169  delay = delay / STX104_CLOCK_RESOLUTION_NS;
170  tsd_reg = (uint32_t) delay;
171 
172  STX104_Index_Write_U32( board, STX104_TRIGGER_START_DELAY, tsd_reg );
173  STX104_Index_Write_U16( board, STX104_TRIGGER_CONFIGURATION, reg_value );
174 
175  return error_code;
176 STX104_Trg_CFG_Set_Error:
177  if ( error_code < 0 ) error_code = -error_code;
178  Error_Log( DEVICE_STX104_NAME, __func__, __LINE__, error_code );
179  return -error_code;
180 }
