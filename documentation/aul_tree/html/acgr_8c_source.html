137  fprintf( out, "/* File: %s */\n", file_name );

154 /**********************************************************************************************//**
155  * Acgr report CSV module.
156  *
157  * \author Mike
158  * \date 3/30/2014
159  *
160  * \param column_names_include true to column names.
161  * \param file_name Filename of the file.
162  *
163  * \return An AULCALLCONV.
164  **************************************************************************************************/
165 AUL_DLL_API int AULCALLCONV ACGR_Report_CSV_Module( size_t column_names_include, const char * file_name )
166 {
167  int error_code;
168  FILE * out;
169 #if ( TARGET_AUTO_CODE_GEN_REPORTS == 1 )
170  MODULE_CSV_COLUMN_ENUM col;
171  size_t index;
172  size_t row;
173  size_t row_max;
174  size_t col_max;
175  char buf[ACGR_BUF_SIZE];
176 #endif
177 
178  out = fopen( file_name, "wt" );
179  if ( NULL == out )
180  { error_code = EC_ACGR_FILE_OPEN_ERROR; goto ACGR_Report_CSV_Module_Error; }
181 
182 #if ( TARGET_AUTO_CODE_GEN_REPORTS == 1 )
183 # ifdef ACGR_PREPEND_TIMESTAMP
184  ACGR_Date_Time( out, file_name, FALSE );
185 # endif
186  row_max = Module_All_Count();
187  col_max = sizeof( acgr_csv_module_columns ) / sizeof ( MODULE_CSV_COLUMN_ENUM );
188 
189  /** column names */
190  if ( column_names_include )
191  {
192  for ( index = 0; index < col_max; index++ )
193  {
194  col = acgr_csv_module_columns[index];
195  error_code = Module_CSV_Column_Name( col, buf, ACGR_BUF_SIZE );
196  if ( error_code )
197  { error_code = EC_ACGR_COLUMN_NAME; goto ACGR_Report_CSV_Module_Error; }
198  if ( index ) fprintf( out, "," );
199  fprintf( out, "\"%s\"", buf );
200  }
201  fprintf( out, "\n" );
202  }
203  /* output a row at a time */
204  for ( row = 0; row < row_max; row++ )
205  { /* rows */
206  for ( index = 0; index < col_max; index++ )
207  { /* columns */
208  col = acgr_csv_module_columns[index];
209  error_code = Module_CSV_Cell_Get( row, col, buf, ACGR_BUF_SIZE );
210  if ( error_code )
211  { error_code = EC_ACGR_CELL_ERROR; goto ACGR_Report_CSV_Module_Error; }
212  if ( index ) fprintf( out, "," );
213  fprintf( out, "\"%s\"", buf );
214  }
215  fprintf( out, "\n" );
216  }
217  fclose( out );
218  return SUCCESS;
219 #else /* TARGET_AUTO_CODE_GEN_REPORTS */
220  fprintf( out, "ACGR_Report_CSV_Module: unavailable. TARGET_AUTO_CODE_GEN_REPORTS != 1\n" );
221  error_code = EC_ACGR_REPORTS_UNAVAILABLE;
222 #endif /* TARGET_AUTO_CODE_GEN_REPORTS */
223  ACGR_Report_CSV_Module_Error:
224  if ( NULL != out ) fclose( out );
225  if ( error_code < 0 ) error_code = -error_code;
226  Error_Log( MODULE_ACGR_NAME, __func__, __LINE__, error_code );
227  return -error_code;
228 }
