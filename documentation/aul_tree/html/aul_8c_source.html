229  if ( !IO_Is_Definition_Available() )
230  { /* Try to acquire from driver -- driver will use acronym

235  if ( error_code ) goto AUL_Device_Open_Error;
236

238  {
239  driver_override = 1;
240  }
241  }
242 
243  if ( AULHwDef_List_Is_Populated() )
244  { /* Use I/O parameters found in the hardware definition file.
245  We will use acronym to find. The acronym must match the

250  if ( error_code ) goto AUL_Device_Open_Error;

252  String_Copy( io_definition_hwdef.acronym, device_acronym_name, IO_DEFINITION_ACRONYM_SIZE );

257  &(io_definition_hwdef.interrupt_number),
258  &(io_definition_hwdef.region_restrict_8bit)
259  );
260  if ( error_code ) goto AUL_Device_Open_Error;
261 #ifdef _WIN32
262  /* reconcile the driver and hardware definition list */
263  if ( driver_override )
264  {
265  if ( 0 == io_definition_driver.region_offset )
266  { io_definition_driver.region_offset = io_definition_hwdef.region_offset; }
267  if ( 0 == io_definition_driver.region_bytes )
268  { io_definition_driver.region_bytes = io_definition_hwdef.region_bytes; }
269  if ( 0 == io_definition_driver.region_restrict_8bit )
270  { io_definition_driver.region_restrict_8bit = io_definition_hwdef.region_restrict_8bit; }
271  if ( 0 == io_definition_driver.interrupt_number )
272  { io_definition_driver.interrupt_number = io_definition_hwdef.interrupt_number; }
273  if ( 0 == io_definition_driver.physical_id )
274  { io_definition_driver.physical_id = io_definition_hwdef.physical_id; }
275  }
276  else
277  { /* hardware definition file is all we got to go on now. */
278  Mem_Copy( &io_definition_driver, &io_definition_hwdef, sizeof( io_definition_driver ) );
279  }
280 #else
281  error_code = EC_AUL_INCOMPLETE_CODE; goto AUL_Device_Open_Error;
282 #endif
283  }
284

289  if ( !IO_Is_Definition_Available() )

335  error_code = Device_Available_Find( io_definition_driver.physical_id, &device_definition );
336  if ( error_code )

338

341  { error_code = EC_AUL_DUPLICATE_DEVICE; goto AUL_Device_Open_Error; }
342 
343 #if(0)

345  if ( 0 == io_definition_driver.region_bytes )

358  device_definition
359  );

371 #if defined( AUL_DEBUG_LOG )
372  Debug_Log( AUL_DEBUG_LOG "%s: exit ec=%s\n", __func__, Status_EC_String( error_code ) );
373 #endif
374  return -error_code;

455 /**********************************************************************************************************//**
456 @ingroup general_libarary
457 \brief
458 
459 
460 \param[in] board The descriptor obtained from the function ::AUL_Device_Open.
461 \param[in] command
462 \param[in] argument
463 
464 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
465 */
466 AUL_DLL_API int AULCALLCONV AUL_Ioctl( int board,
467  unsigned int command,
468  unsigned long argument
469  )
470 {
471  int error_code;
472 /* size_t type;*/
473  size_t cmd;
474  size_t cmd_set;
475  size_t index; /* or channel */
476  /*size_t instance;*/
477  IOC_T physical_id; /* NOTE: modules and devices: logical_id is the same as physical_id */
478  devmgr_device_type * d; /* short hand for device */
479 
480 #if defined( IO_DEBUG_LOG )
481  Debug_Log( AUL_DEBUG_LOG "%s: entry\n", __func__ );
482 #endif
483 /*
484 Order of communication:
485 1. lib
486 2. driver
487 3. device
488 4. component
489 */
490  /* If intended for driver top level, then pass it along. */
491  if ( AULDRV_LOGICAL_ID == IOC_GET_LOGICAL_ID_MODULE( command ) )
492  {
493  error_code = IO_Ioctl( board, command, argument );
494  goto AUL_Ioctl_Finish;
495  }
496 
497  if ( SUCCESS != ( error_code = AUL_Board( board, &d ) ) ) goto AUL_Ioctl_Error;
498  cmd_set = 0;
499  cmd = 0;
500  /* test for validity of the command */
501  if ( IOC_MAGIC_IS_VALID_LIB(command) )
502  {
503  cmd_set = 1;
504  cmd = IOC_GET_COMMAND( command );
505  }
506  else if ( IOC_MAGIC_IS_VALID( command ) )
507  { /* pass onto driver */
508  error_code = IO_Ioctl( aul_device[board]->io_device, command, argument );
509  if ( error_code ) goto AUL_Ioctl_Error;
510  }
511  else
512  { error_code = EC_AUL_IOCTL_COMMAND; goto AUL_Ioctl_Error; }
513  /* test validity of the device */
514  if ( NULL == d )
515  { error_code = EC_AUL_DEVICE_NULL_POINTER; goto AUL_Ioctl_Error; }
516  if ( 0 == d->open_and_active )
517  { error_code = EC_AUL_DEVICE_NOT_OPEN; goto AUL_Ioctl_Error; }
518  /* obtain information from command */
519  physical_id = IOC_GET_PHYSICAL_ID( command );
520  index = IOC_GET_INDEX( command );
521  /*instance = IOC_GET_INSTANCE( command );*/
522 
523  if ( IOC_VALID_LOGICAL_ID_MODULE( physical_id /* OK since only important parts tested */ ) )
524  { /* this is a module command */
525  if ( IO_LOGICAL_ID == physical_id )
526  {
527  error_code = IO_Ioctl( aul_device[board]->io_device, command, argument );
528  if ( error_code ) goto AUL_Ioctl_Error;
529  }
530  else
531  { error_code = EC_AUL_IOCTL_COMMAND; goto AUL_Ioctl_Error; }
532  }
533  else if ( IOC_VALID_LOGICAL_ID_DEVICE( physical_id /* OK since only import parts tested */ ) )
534  { /* this is a device command */
535  if ( !cmd_set ) { error_code = EC_AUL_IOCTL_COMMAND; goto AUL_Ioctl_Error; }
536  /* error_code = ( * d->definition->jump_table[cmd] )( d, index, (char *) argument, 0 ); */
537  error_code = DevMgr_Command( d, cmd, index, ( char *) argument, 0 );
538  }
539  else
540  { /* component command -- assume so, anyway */
541  int cmp_table_index;
542  component_type * component; /* short hand */
543  component_type * cmp_table;
544 /*TODO: add debugging macro around this: #if defined()...*/
545  if ( !cmd_set ) { error_code = EC_AUL_IOCTL_COMMAND; goto AUL_Ioctl_Error; }
546  if ( NULL == d->component_table ) { error_code = EC_AUL_IOCTL_COMPONENT; goto AUL_Ioctl_Error; }
547 /*end*/
548  cmp_table = d->component_table;
549  /* must search to find the component */
550  cmp_table_index = CmpMgr_Component_Match_Physical_ID( cmp_table, physical_id );
551  if ( CMPMGR_EC_FAILURE == cmp_table_index )
552  { error_code = EC_AUL_IOCTL_COMPONENT; goto AUL_Ioctl_Error; }
553  component = &( cmp_table[cmp_table_index] );
554  error_code = CmpMgr_Command( component, /* info about the component */
555  cmd, /* component command for jump table */
556  index, /* channel within component */
557  (char *) argument,
558  0
559  );
560  }
561 AUL_Ioctl_Finish:

764 /**********************************************************************************************************//**
765 @ingroup general_libarary
766 \brief
767 Apex Universal Library function that returns
768 
769 \param[out] device_list
770 \return An error code of type int is returned. If successful, returns 0 (SUCCESS), otherwise it returns a non-zero error code. Please refer to ::AUL_STATUS_ENUM for all the possible error codes.
771 */
772 AUL_DLL_API int AULCALLCONV AUL_Device_List_Get( device_list_type * device_list )
773 {
774  int error_code;
775 #if defined( IO_DEBUG_LOG )
776  Debug_Log( AUL_DEBUG_LOG "%s: entry\n", __func__ );
777 #endif
778  error_code = AUL_Ioctl( -1 /* controller */, CMD_AULDRV_DEVICE_LIST_GET, (unsigned long) device_list );
779  if ( error_code < 0 ) error_code = -error_code;
780  if ( EC_IO_WARN_DEVICE_LIST_UNKNOWN == error_code )
781  { /* retrieve the information from hardware definition file */
782  size_t index;
783  Mem_Set( device_list, 0, sizeof( *device_list ) );
784  device_list->type = SYS_TYPE_MAKE( device_list_type );
785  for ( index = 0; index < DEVICE_QTY; index++ )
786  { /* add in the open and active status */
787  error_code = AULHwDef_Parameters_Using_Index_Get( index,
788  device_list->region[index].acronym,
789  &(device_list->region[index].physical_id), /* actually logical id */
790  &(device_list->region[index].region_offset),
791  &(device_list->region[index].region_bytes),
792  &(device_list->region[index].interrupt_number),
793  &(device_list->region[index].region_restrict_8bit)
794  );
795  if ( error_code ) break;
796  device_list->region_count++;
797  }
798  error_code = EC_IO_WARN_DEVICE_LIST_PARTIAL; /* to perform subsequent if-then statement below */
799  }
800  if ( SYS_TYPE_MAKE( device_list_type ) != device_list->type )
801  { error_code = EC_AUL_DATA_TYPE_ERROR; goto AUL_Device_List_Get_Error; }
802  if ( EC_IO_WARN_DEVICE_LIST_PARTIAL == error_code )
803  { /* need to add open status information */
804  int index, count;
805  for ( index = 0; index < DEVICE_QTY; index++ )
806  { /* add in the open and active status */
807  if ( NULL != aul_device[index] )
808  { /* loop through io_definitions to find a match */
809  for ( count = 0; count < device_list->region_count; count++ )
810  {
811  if ( aul_device[index]->physical_id == device_list->region[count].physical_id )
812  { /* we have a match -- now assign library open status */
813  device_list->open_and_active[count] = aul_device[index]->open_and_active;
814  break;
815  }
816  }
817  }
818  }
819  error_code = SUCCESS;
820  }
821  if ( error_code ) goto AUL_Device_List_Get_Error;
822 #if defined( IO_DEBUG_LOG )
823  Debug_Log( AUL_DEBUG_LOG "%s: exit success\n", __func__ );
824 #endif
825  return error_code;
826 AUL_Device_List_Get_Error:
827  if ( error_code < 0 ) error_code = -error_code;
828  Error_Log( MODULE_AUL_NAME, __func__, __LINE__, error_code );
829 #if defined( AUL_DEBUG_LOG )
830  Debug_Log( AUL_DEBUG_LOG "%s: exit ec=%s\n", __func__, Status_EC_String( error_code ) );
831 #endif
832  return -error_code;
833 }

835 AUL_DLL_API int AULCALLCONV AUL_Device_Logical_Get( const char * device_acronym_name, DEVICE_LOGICAL_ENUM * logical_id )
836 { /* NOTICE: this code is EXTREMELY similar to the AUL_Device_Open(). TODO: merge/reduce */
837  int error_code;
838  device_definition_type * device_definition;
839  io_definition_type io_definition;
840 #if defined( MODULE_AULHWDEF )
841  void * hwdef;
842 #endif
843 #if defined( IO_DEBUG_LOG )
844  Debug_Log( AUL_DEBUG_LOG "%s: entry\n", __func__ );
845 #endif
846  Mem_Set( &io_definition, 0, sizeof( io_definition ) );
847  error_code = String_Copy( io_definition.acronym, device_acronym_name, IO_DEFINITION_ACRONYM_SIZE );
848  if ( ( IO_DEFINITION_ACRONYM_SIZE == error_code ) || ( 1 >= error_code ) )
849  { error_code = EC_AUL_ACRONYM_SIZE; goto AUL_Device_Logical_Get_Error; }
850 
851  if ( !IO_Is_Definition_Available() )
852  {
853  error_code = IO_Definition_Get( &io_definition ); /* returns data based on acronym */
854  if ( error_code ) goto AUL_Device_Logical_Get_Error;
855 
856  if ( 0 == String_Compare( io_definition.acronym, device_acronym_name, IO_DEFINITION_ACRONYM_SIZE ) )
857  {
858  goto AUL_Device_Logical_Get_Finish;
859  }
860  }
861  else if ( AULHwDef_List_Is_Populated() )
862  { /* Use I/O parameters found in the hardware definition file.
863  We will use acronym to find. The acronym must match the
864  file node name as well. We want that to be the case so
865  that things are consistent across all platforms.
866  */
867  error_code = AULHwDef_Acronym_Find( device_acronym_name, &(io_definition.minor_number), &hwdef );
868  if ( error_code ) goto AUL_Device_Logical_Get_Error;
869  /* we must have a match, so copy it to destination */
870  error_code = AULHwDef_Parameters_Get( hwdef,
871  &(io_definition.physical_id), /* actually logical id */
872  &(io_definition.region_offset),
873  &(io_definition.region_bytes),
874  &(io_definition.interrupt_number),
875  &(io_definition.region_restrict_8bit)
876  );

888  if ( AUL_Device_IsDuplicate( &io_definition ) )

891  return error_code;
892 AUL_Device_Logical_Get_Error:
