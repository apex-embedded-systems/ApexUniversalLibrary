187 {
188  const char * string_pointer;
189  size_t string_length;
190 #define COMMAND_CW_LEN 32
191  char buf[COMMAND_CW_LEN];
192 
193  size_t status_count = Command_All_Count();
194 
195  if ( row >= status_count ) goto Command_CSV_Cell_Get_Error;
196 
197  if ( ( column >= COMMAND_CSV_COLUMN_UNKNOWN ) ||
198  ( column < COMMAND_CSV_COLUMN_ENUM_VALUE ) ) goto Command_CSV_Cell_Get_Error;
199 
200  /* initialize */
201  Mem_Set( cell_buf_io, '\0', cell_buf_size );
202  string_length = 0;
203  string_pointer = NULL;
204  switch( column )
205  {
206  case COMMAND_CSV_COLUMN_ENUM_VALUE:
207  Mem_Set( buf, 0, COMMAND_CW_LEN );
208  CMD_SNPRINTF( buf, COMMAND_CW_LEN, "%d", command_ds_list[row].cmd );
209  if ( '\0' != buf[COMMAND_CW_LEN - 1] )
210  { buf[COMMAND_CW_LEN - 1] = '\0'; goto Command_CSV_Cell_Get_Error; }
211  string_pointer = buf;
212  break;
213  case COMMAND_CSV_COLUMN_ENUM_SYMBOL:
214  string_pointer = command_ds_list[row].cmd_name;
215  break;
216  /* case COMMAND_CSV_COLUMN_MESSAGE:
217  string_pointer = command_ds_list[row].name;
218  break; */
219  default: break;
220  }
221  if ( NULL == string_pointer ) return SUCCESS;
222  string_length += String_Length( string_pointer ) + 1 /* null */;
223  String_Concatenate( cell_buf_io, string_pointer, cell_buf_size );
224  if ( string_length >= cell_buf_size ) goto Command_CSV_Cell_Get_Error;
225  return SUCCESS;
226 Command_CSV_Cell_Get_Error:
