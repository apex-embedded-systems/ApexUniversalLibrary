335 /*****************************************************************
336 Description
337 
338 Most likely this will be used as part of the device initialization process.
339 According to our prototype, devices need a Load() and Init() functions.
340 
341 Brackets in the parameter list to remind us that it is a null terminated list
342 
343 For the component_list[] it is possible to pass a partial list. These will be verified first
344 that they exist in the available_list[]. It is assumed that the component_list[] has already
345 been allocated; specifically, the definition, physical_id, and private data have been assigned.
346 It is also assumed that the component_list[] has sufficient room for all the components
347 discovered during initialization (i.e. what is in the inventory_list[]).
348 
349 The inventory_list[] can be built during hardware discovery or hard-coded within the device
350 files. If the inventory_list[] is null, then it is assumed that only offset and component_link
351 fixups are required. These fixups can be avoided by status=CMP_STS_BUILD(1,0,0,0,0);
352 If inventory_list[] is null, it will only apply offset fixups to direct addressing registers
353 which is all that there can be.
354 
355 if component_link is not null in the component_list[], then it is assumed to have been fixed up.
356 It will make sure that the status value is updated appropriately.
357 
358 
359 */
360 int CmpMgr_List_Setup( component_inventory_type inventory_list[], /* INPUT -- enumerated inventory found in h/w or predefined for device */
361  size_t direct_offset, /* INPUT -- base address of device */
362  int component_qty, /* INPUT -- output quantity limit */
363  component_type component_list[] /* OUTPUT -- device component list, some optionally defined. */
364  )
365 {
366  int error_code;
367  int inv_index;
368  int avl_index;
369  int cmp_index;
370  size_t iampid; /* short for inventory address mode physical ID */
371  IOC_T inv_logical_id;
372  IOC_T cmp_logical_id;
373  component_type * cmp; /* short hand for component_list[cmp_index] */
374  component_inventory_type * inv; /* short hand for inventory_list[inv_index] */
375 
376  /* verify integrity of input parameters */
377  error_code = SUCCESS;
378 /* if ( NULL == component_available_list )
379  { error_code = EC_CMPMGR_SETUP_AVAILABLE_LIST; goto CmpMgr_Device_Setup_Error; }
380  if ( NULL == inventory_list )
381  { error_code = EC_CMPMGR_SETUP_INVENTORY_LIST; goto CmpMgr_Device_Setup_Error; }*/
382  if ( NULL == component_list )
383  { error_code = EC_CMPMGR_SETUP_COMPONENT_LIST; goto CmpMgr_Device_Setup_Error; }
384  if ( 0 == component_qty )
385  { error_code = EC_CMPMGR_SETUP_COMPONENT_QTY; goto CmpMgr_Device_Setup_Error; }
386 
387  /* Verify that all components already specified in component_list[] are found in
388  availability list.
389  */
390  cmp_index = 0;
391  cmp_logical_id = IOC_GET_LOGICAL_ID_COMPONENT( component_list[cmp_index].physical_id );
392  while ( IOC_PHYSICAL_ID_NULL != cmp_logical_id )
393  {
394  avl_index = CmpMgr_Available_Match_Logical_ID( component_available_list, cmp_logical_id );
395  if ( CMPMGR_EC_FAILURE == avl_index )
396  { error_code = EC_CMPMGR_SETUP_AVAILABLE; goto CmpMgr_Device_Setup_Error; }
397 
398  cmp_index++;
399  if ( cmp_index >= component_qty )
400  { error_code = EC_CMPMGR_SETUP_COMPONENT_QTY; goto CmpMgr_Device_Setup_Error; }
401  cmp_logical_id = IOC_GET_LOGICAL_ID_COMPONENT( component_list[cmp_index].physical_id );
402  }
403 
404  if ( NULL != inventory_list )
405  { /* At this point cmp_index is at the next available element.
406  Now, instantiate the components from inventory list into
407  device component list.
408  */
409  /* verify that inventory has at least one item in the list */
410  inv_index = 0;
411  inv_logical_id = inventory_list[inv_index].logical_id;
412  if ( IOC_PHYSICAL_ID_NULL == inv_logical_id )
413  { error_code = EC_CMPMGR_SETUP_INVENTORY; goto CmpMgr_Device_Setup_Error; }
414 
415  inv_index = 0;
416  do
417  { /* find a logical ID match between what is available and the inventory */
418  avl_index = CmpMgr_Available_Match_Logical_ID( component_available_list, inv_logical_id );
419  if ( CMPMGR_EC_FAILURE == avl_index )
420  { error_code = EC_CMPMGR_SETUP_AVAILABLE; goto CmpMgr_Device_Setup_Error; }
421  /* map parameters into the device level component */
422  cmp = &(component_list[cmp_index]);
423  inv = &(inventory_list[inv_index]);
424  cmp->definition = component_available_list[avl_index].definition;
425  cmp->physical_id = IOC_CREATE_PHYSICAL_ID_COMPONENT( inv->logical_id, inv->instance );
426 /*TODO: check for private data already fixed up???????*/
427  cmp->private_data = inv->private_data;
428  cmp->status = CMP_STS_SET_PRIVATE_DATA_FXU( cmp->status, CMP_STS_PRIVATE_DATA_FXU_ON );
429  cmp->offset = inv->offset;
430 /* fixup direct addressing
431  if ( IOC_PHYSICAL_ID_NULL == inv->address_mode_physical_id )
432  { DIRECT IO
433  cmp->component_link = NULL;
434  cmp->offset += direct_offset;
435  cmp->status = CMP_STS_SET_OFFSET_FXU( cmp->status, CMP_STS_OFFSET_FXU_ON );
436  } */
437  /* NOTE: INDIRECT IO addressing will be fixed up at the end of this function */
438  inv_index++; cmp_index++;
439  inv_logical_id = inventory_list[inv_index].logical_id;
440  if ( cmp_index >= component_qty )
441  { error_code = EC_CMPMGR_SETUP_LIST_FULL; goto CmpMgr_Device_Setup_Error; }
442  } while ( IOC_PHYSICAL_ID_NULL != inv_logical_id );
443  }
444  /* Post-addressing-mode-fix-ups. This includes both direct and indirect addressing.
445  Any component that did not get an offset fix up will be subject to addressing
446  mode fixups.
447  */
448  cmp_index = 0;
449  cmp = &(component_list[cmp_index]);
450  while ( IOC_PHYSICAL_ID_NULL != cmp->physical_id )
451  {
452  if ( NULL == cmp->component_link )
453  { /* not fixed up, so green light to go further with fix-ups */
454  if ( CMP_STS_OFFSET_FXU_OFF == CMP_STS_GET_OFFSET_FXU( cmp->status ) )
455  { /* offset and address mode confirmed to not have been fixed up */
456  if ( NULL == inventory_list )
457  { /* can only be direct addressing scheme */
458  iampid = IOC_PHYSICAL_ID_NULL; /* force direct addressing fix ups */
459  }
460  else
461  { /* find in the inventory where this component is located to obtain address mode physical id */
462  inv_index = CmpMgr_Inventory_Match_Physical_ID( inventory_list, cmp->physical_id );
463  iampid = inventory_list[inv_index].address_mode_physical_id;
464  }
465  if ( IOC_PHYSICAL_ID_NULL == iampid )
466  { /* direct addressing */
467  cmp->component_link = NULL;
468  cmp->offset += direct_offset;
469  }
470  else
471  { /* indirect addressing */
472  /* NOTE: re-using avl_index as just a general index */
473  avl_index = CmpMgr_Component_Match_Physical_ID( component_list, iampid );
474  if ( CMPMGR_EC_FAILURE == avl_index )
475  { error_code = EC_CMPMGR_SETUP_ADDRESSING; goto CmpMgr_Device_Setup_Error; }
476  cmp->component_link = &(component_list[avl_index]);
477  }
478  cmp->status = CMP_STS_SET_OFFSET_FXU( cmp->status, CMP_STS_OFFSET_FXU_ON );
479  }
480  }
481  else
482  { /* component_link already fixed up so lets make sure that the status follows suit */
483  if ( CMP_STS_OFFSET_FXU_OFF == CMP_STS_GET_OFFSET_FXU( cmp->status ) )
484  {
485  cmp->status = CMP_STS_SET_OFFSET_FXU( cmp->status, CMP_STS_OFFSET_FXU_ON );
486  }
487  }
488  cmp_index++;
489  if ( cmp_index >= component_qty )
490  { error_code = EC_CMPMGR_SETUP_LIST_FULL; goto CmpMgr_Device_Setup_Error; }
491  cmp = &(component_list[cmp_index]);
492  }
493  return SUCCESS;
494 
495 CmpMgr_Device_Setup_Error:
496  Error_Log( MODULE_CMP_MGR_NAME, __func__, __LINE__, error_code );
497  return -error_code;
498 }
