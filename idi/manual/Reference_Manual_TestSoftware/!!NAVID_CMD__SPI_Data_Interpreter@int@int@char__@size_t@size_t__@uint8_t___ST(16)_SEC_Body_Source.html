35: 		}
36: 		else
37: 		{

39: 			index_byte++;
40: 			sp = NULL;

43: 	}
44: 
45: 	if ( count > 0 )
46: 	{
47: #if defined( __MSDOS__ )
48: 		printf( "Warning: ignored %d arguments", count );
49: #else

51: #endif
52: 		if ( NULL != sp )
53: 		{
54: 			printf( ", and portion of string argument" );
55: 		}
56: 		printf( "\n" );
57: 	}
58:
