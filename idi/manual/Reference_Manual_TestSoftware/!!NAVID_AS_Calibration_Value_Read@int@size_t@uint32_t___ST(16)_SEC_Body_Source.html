 1: static int AS_Calibration_Value_Read( int board_id, size_t address, uint32_t * value )
 2: {
 3: 	int			error_code;
 4: 	uint32_t	value_temp;
 5: 	size_t 		index;
 6: 	size_t		object_count;
 7: 	size_t		count      = 3;
 8: 	uint8_t		tx_buf[12] = { 0x00 };
 9: 	uint8_t		rx_buf[12] = { 0x00 };
10: 
11: #define FUNCTION_NAME__  "AS_Calibration_Value_Read"
12: #if ( IO_TRACE_USE_AS_LOGGING == 1 )
13: 	IO_Trace_Log_Function( FUNCTION_NAME__, __LINE__, IO_TRACE_AS, board_id, 0 );
14: #endif
15: #undef FUNCTION_NAME__
16: 
17: 	tx_buf[0] = 0x20;
18: 	
19: 	address = 0x0F & address;
20: 	tx_buf[0] = tx_buf[0] + ( (uint8_t) address );
21: 	tx_buf[1] = tx_buf[1] + ( (uint8_t) count   );
22: 	
23: 	object_count = count + 2;
24: 	AS_Chip_Select_Route_Override( board_id );
25: 	SPI_Configuration_Chip_Select_Behavior_Set( board_id, CSB_BUFFER );
26: 	error_code =  SPI_Data_Write_Read(  board_id,		
27: 										1,				
28: 										object_count,	
29: 										tx_buf, 		
30: 										object_count,	
31: 										rx_buf 			
32: 							  	  	  );
33: 	if ( error_code < 0 ) goto AS_Calibration_Value_Read_Error_Exit;
34: 	value_temp = 0;
35: 	object_count--;
36: 	for ( index = 0; index < count; index++ ) value_temp = ( value_temp << 8 ) + ( (uint32_t) rx_buf[object_count - index] );
37: 	*value = value_temp;
38: 	AS_Chip_Select_Route_Restore( board_id );
39: 	return SUCCESS;
40: AS_Calibration_Value_Read_Error_Exit:
41: 	AS_Chip_Select_Route_Restore( board_id );
42: 	return error_code;
43: }
