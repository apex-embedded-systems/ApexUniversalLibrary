 1: int SPI_Report_Status_Text( struct spi_status * status, FILE * out )
 2: {
 3: 	fprintf( out, "################################### SPI " );

 6:

11: 	fprintf( out, "fifo size  = %d\n", status->fifo_size                );
12: 	fprintf( out, "fifo count = %d\n", status->fifo_count               );
13: 
14: 	return SUCCESS;
15: }
